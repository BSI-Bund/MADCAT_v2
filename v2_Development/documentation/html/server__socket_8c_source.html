72 void on_client_read(void* closure, char* buffer, int len)
73 {
74  struct proxy_data* data = (struct proxy_data*) closure;
75  if (data->backend == NULL) {
76  return;
77  }
78  connection_write(data->backend, buffer, len);
79  //MADCAT
80  //log, using data->client as id, which also contains the struct epoll_event_handler*
81  long double unix_timeasdouble = time_str(NULL, 0, NULL, 0);
82  struct json_data_node_t* jd_node = jd_get(jd, (uintptr_t ) data->client);
83  jd_node->bytes_toserver += len;
84 
85  if( !jd_node->firstpacket && (unix_timeasdouble - jd_node->last_recv < jd_node->min_rtt || jd_node->min_rtt == 0 ) ) {
86  jd_node->min_rtt = unix_timeasdouble - jd_node->last_recv;
87  //fprintf(stderr, "\n*****DEBUG: min_rtt: %Lf\n", jd_node->min_rtt);
88  }
89 
90  jd_node->last_recv = unix_timeasdouble;
91  jd_node->firstpacket = false;
92 }

143 struct proxy_data* handle_client_connection(int client_socket_fd,
144  char* backend_host,
145  char* backend_port_str)
146 {
147  struct epoll_event_handler* client_connection;
148  rsp_log("Creating connection object for incoming connection...");
149  client_connection = create_connection(client_socket_fd);
150 
151  int backend_socket_fd = connect_to_backend(backend_host, backend_port_str);
152  struct epoll_event_handler* backend_connection;
153  rsp_log("Creating connection object for backend connection...");
154  backend_connection = create_connection(backend_socket_fd);
155 
156  struct proxy_data* proxy = malloc(sizeof(struct proxy_data));
157  proxy->client = client_connection;
158  proxy->backend = backend_connection;
159 
160 
161  //MADCAT start
162  if ( !jd_get(jd, (uintptr_t ) proxy->client)) jd_push(jd, (uintptr_t ) proxy->client);
163  struct json_data_node_t* jd_node = jd_get(jd, (uintptr_t ) proxy->client);
164 
165  jd_node->bytes_toclient = 0;
166  jd_node->bytes_toserver = 0;
167 
168  //proxy->bytes_toclient = 0; //MADCAT
169  //proxy->bytes_toserver = 0; //MADCAT
170 
171 
172  //Get local client address and port
173  struct sockaddr local_address;
174  socklen_t addr_size = sizeof(local_address);
175  getsockname(backend_socket_fd, &local_address, &addr_size);
176 
177  char* port_ptr = local_address.sa_data;
178  char* ip_ptr = (char*) &(local_address.sa_data) + 2;
179  proxy_sock.client_port = (uint16_t) ((uint8_t) (*port_ptr)) * 256 + ((uint8_t) (*(port_ptr+1)));
180  //proxy_sock.client_addr = inttoa(*(uint32_t*)ip_ptr); //Commented out, what was my thought?
181 
182  jd_node->proxy_ip = inttoa(*(uint32_t*)ip_ptr);
183  jd_node->proxy_port = proxy_sock.client_port;
184  jd_node->backend_ip = strncpy(malloc(strlen(proxy_sock.backend_addr) +1 ), proxy_sock.backend_addr, strlen(proxy_sock.backend_addr) +1 );
185  jd_node->backend_port = strncpy(malloc(strlen(proxy_sock.backend_port_str) +1 ), proxy_sock.backend_port_str, strlen(proxy_sock.backend_port_str) +1 );
186 
187  //MADCAT end
188 
189  struct connection_closure* client_closure = (struct connection_closure*) client_connection->closure;
190  client_closure->on_read = on_client_read;
191  client_closure->on_read_closure = proxy;
192  client_closure->on_close = on_client_close;
193  client_closure->on_close_closure = proxy;
194 
195  struct connection_closure* backend_closure = (struct connection_closure*) backend_connection->closure;
196  backend_closure->on_read = on_backend_read;
197  backend_closure->on_read_closure = proxy;
198  backend_closure->on_close = on_backend_close;
199  backend_closure->on_close_closure = proxy;
200 
201  return proxy; //MADCAT
202 }

352 struct epoll_event_handler* create_server_socket_handler(char* server_addr,
353  char* server_port_str,
354  char* backend_addr,
355  char* backend_port_str)
356 {
357 
358  int server_socket_fd;
359  server_socket_fd = create_and_bind(server_addr, server_port_str);
360  make_socket_non_blocking(server_socket_fd);
361 
362  listen(server_socket_fd, MAX_LISTEN_BACKLOG);
363 
364  struct server_socket_event_data* closure = malloc(sizeof(struct server_socket_event_data));
365  closure->backend_addr = backend_addr;
366  closure->backend_port_str = backend_port_str;
367 
368  struct epoll_event_handler* result = malloc(sizeof(struct epoll_event_handler));
369  result->fd = server_socket_fd;
370  result->handle = handle_server_socket_event;
371  result->closure = closure;
372 
373  epoll_add_handler(result, EPOLLIN | EPOLLET);
374 
375  return result;
376 }
