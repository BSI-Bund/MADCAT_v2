101 int get_config_table(lua_State* L, char* name, struct proxy_conf_udp_t* pc) //read proxy configuration from parsed LUA-File by luaL_dofile(...). Returns number of read elements.
102 {
103  char* backendaddr = 0;
104  int backendport = 0;
105 
106  int num_elements = 0;
107 
108  lua_getglobal(L, name); //push objekt "name" to stack and...
109 
110  if ( !lua_istable(L, -1) ) { //...check if this objekt is a table
111  fprintf(stderr, "\tNo proxy config found. Variable \"%s\" must be a LUA table.\n", name);
112  return num_elements;
113  }
114 
115  //Iterate over all possible portnumbers.
116  //Think about a more clever solution?
117  for (int listenport = 0; listenport<65536; listenport++) {
118  lua_pushnumber(L, listenport); //push actuall portnumber on stack and...
119  lua_gettable(L, -2); //...call lua_gettable with this portnumber as key
120  if( !lua_isnil(L,-1) ) { //if corresponding value is not NIL...
121  lua_pushnumber(L, 1); //push "1" on the stack for the first elemnt in sub-table and...
122  lua_gettable(L, -2); //...fetch this entry
123  backendaddr = (char*) lua_tostring(L, -1);
124  lua_pop(L, 1); //remove result from stack
125 
126  lua_pushnumber(L, 2); //push "2" on the stack for the second elemnt in sub-table and...
127  lua_gettable(L, -2); //...fetch this entry
128  backendport = lua_tonumber(L, -1);
129  lua_pop(L, 1); //remove result from stack
130 
131  pcudp_push(pc, listenport, backendaddr, backendport);
132  pc->portmap[listenport] = true;
133  num_elements++;
134  }
135  lua_pop(L, 1); //remove sub-table from stack
136  }
137  return num_elements;
138 }

214 udpcon_id_t* uc_genlid(char* src_ip, uint64_t src_port, char* dest_ip, uint64_t dest_port, udpcon_id_t* output)
215 {
216  udpcon_id_t* id = 0;
217  uint64_t id_src = 0;
218  uint64_t id_dest = 0;
219  struct sockaddr_in src_sa;
220  struct sockaddr_in dest_sa;
221 
222  if(output == NULL) id = malloc(sizeof(udpcon_id_t));
223  else id = output;
224 
225  inet_pton(AF_INET, src_ip, &(src_sa.sin_addr));
226  inet_pton(AF_INET, dest_ip, &(dest_sa.sin_addr));
227  //Concatinate IPs and Ports.
228  /*Shifting portnumbers to higher bits makes IDs most times (client src_port > backend dest_port) easier to distinguish,
229  thus better human readable, if multiple connections from one IP occur*/
230  id_src = (uint64_t) src_port << 32 | src_sa.sin_addr.s_addr;
231  id_dest = (uint64_t) dest_port << 32 | dest_sa.sin_addr.s_addr;
232  if(id_src > id_dest) { //Make Comparable
233  id->high = id_src;
234  id->low = id_dest;
235  } else {
236  id->low = id_src;
237  id->high = id_dest;
238  }
239 
240  if(output == NULL) {
241  id->str = uc_strlid(id, NULL);
242  id->malloced = true;
243  } else {
244  uc_strlid(id, id->__str);
245  id->str = id->__str;
246  id->malloced = false;
247  }
248 
249  id->masked_id = id->high ^ id->low ^ sessionkey;
250 
251  return id;
252 }

292

297 
298  //DATA
299  uc_node->src_ip = EMPTY_STR;
300  uc_node->src_port = 0;
301  uc_node->dest_ip = EMPTY_STR;
302  uc_node->dest_port = 0;

338 bool uc_del(struct udpcon_data_t* uc, udpcon_id_t id)
339 {
340  //re-implentation of uc_get to archive thread safeness (may not be needed in every case, but saves a function call anyway)
341  struct udpcon_data_node_t* uc_node = uc->list;
342  while ( uc_node != 0) {
343  if(uc_node = uc_get(uc, id)) break;
344  uc_node = uc_node->next;
345  }
346  if (uc_node == 0) return false;
347 
348  //Close sockets
349  if(uc_node->client_socket_fd != 0) close(uc_node->client_socket_fd);
350  if(uc_node->backend_socket_fd != 0) close(uc_node->backend_socket_fd);
351 
352  //Free
353  if (uc_node->backend_socket != NULL) free(uc_node->backend_socket);
354  if (uc_node->client_socket != NULL) free(uc_node->client_socket);
355  if (uc_node->src_ip != EMPTY_STR) free(uc_node->src_ip);
356  if (uc_node->dest_ip != EMPTY_STR) free(uc_node->dest_ip);
357  if (uc_node->timestamp != EMPTY_STR) free(uc_node->timestamp);
358  if (uc_node->start != EMPTY_STR) free(uc_node->start);
359  if (uc_node->end != EMPTY_STR) free(uc_node->end);
360  if (uc_node->proxy_ip != EMPTY_STR) free(uc_node->proxy_ip);
361  if (uc_node->backend_ip != EMPTY_STR) free(uc_node->backend_ip);
362  if (uc_node->payload != NULL) free(uc_node->payload);
363  if (uc_node->first_dgram != NULL) free(uc_node->first_dgram);
364  if(uc_node->id_fromclient.malloced) free(uc_node->id_fromclient.str);
365  if(uc_node->id_tobackend.malloced) free(uc_node->id_tobackend.str);
366 
367  //Rearange pointers
368  if (uc_node == uc->list) uc->list = uc_node->next;
369  if (uc_node->prev != NULL) uc_node->prev->next = uc_node->next;
370  if (uc_node->next != NULL) uc_node->next->prev = uc_node->prev;
371 
372  //Free the list element itself
373  free(uc_node);
374  return true;
375 }

377 int uc_cleanup(struct udpcon_data_t* uc, long long int timeout)
378 {
379  struct timespec sem_timeout; //time to wait in sem_timedwait() call
380  clock_gettime(CLOCK_REALTIME, &sem_timeout);
381  sem_timeout.tv_sec += 1;
382  sem_timedwait(conlistsem, &sem_timeout);
383  struct udpcon_data_node_t* uc_node = uc->list;
384  char unix_time_str[64] = "";
385  time_str(unix_time_str, sizeof(unix_time_str), NULL, 0);
386  long long int unix_time = atoll(unix_time_str);
387  int num_removed = 0;
388  struct udpcon_data_node_t* next = NULL;
389 
390  while ( uc_node != NULL ) {
391  next = uc_node->next; //save next pointer, because the element might be deleted.
392  if(uc_node->last_seen + timeout < unix_time) {
393  sem_post(conlistsem); //spare out the call to json_out, calculation of SHA1 over payloads may take quite some time.
394  json_out(uc_node);
395  clock_gettime(CLOCK_REALTIME, &sem_timeout);
396  sem_timeout.tv_sec += 1;
397  sem_timedwait(conlistsem, &sem_timeout);
398  if(uc_del(uc, uc_node->id_fromclient)) {
399  num_removed++;
400  } else {
401  sem_post(conlistsem);
402  return -1;
403  }
404  }
405  uc_node = next; //Next pointer always contains an element, regardless of main thread, because only here in this function uc_del is called
406  }
407  sem_post(conlistsem);
408  return num_removed;
409 }

473 void json_out(struct udpcon_data_node_t* uc_node)
474 {
475  char* payload_hd_str = "NONE"; //Payload as string in HexDump Format
476  char* payload_str = "NONE"; //Payload as string
477  char * payload_sha1_str = "NONE"; //Paylod SHA1 hash
478  unsigned char payload_sha1[SHA_DIGEST_LENGTH]; //SHA1 of payload
479 
480  //Begin JSON output and open new JSON
481  //Log connection to STDOUT in json-format (Suricata-like)
482  json_do(true, "{\"origin\":\"MADCAT\", \
483 \"src_ip\":\"%s\", \
484 \"dest_port\": %d, \
485 \"timestamp\":\"%s\", \
486 \"unixtime\": %lld, \
487 \"dest_ip\":\"%s\", \
488 \"src_port\": %d, \
489 \"proto\":\"%s\", \
490 \"event_type\":\"%s\", \
491 \"FLOW\": { \
492 \"start\":\"%s\", \
493 \"end\":\"%s\", \
494 \"duration\":%Lf, \
495 \"min_rtt\":%Lf, \
496 \"bytes_toserver\": %lld, \
497 \"bytes_toclient\": %lld\
498 ",\
499  uc_node->src_ip,\
500  uc_node->dest_port,\
501  uc_node->timestamp,\
502  uc_node->unixtime,\
503  uc_node->dest_ip,\
504  uc_node->src_port,\
505  "UDP",\
506  uc_node->proxied ? "proxy_flow" : "flow",\
507  uc_node->start,\
508  uc_node->end,\
509  uc_node->duration,\
510  uc_node->min_rtt,\
511  uc_node->bytes_toserver,\
512  uc_node->bytes_toclient\
513  );
514 
515  if( uc_node->proxied ) { //Proxy specific JSON output
516  json_do(false, ",\
517 \"proxy_ip\":\"%s\", \
518 \"proxy_port\": %d, \
519 \"backend_ip\":\"%s\",\
520 \"backend_port\": %d\
521 ",\
522  uc_node->proxy_ip,\
523  uc_node->proxy_port,\
524  uc_node->backend_ip,\
525  uc_node->backend_port\
526  );
527  }
528  //Do only include payload and compute sha1, if this was not a connection handled by proxy.
529  //Overhead might easily become too large and it is intended to be logged and processed by backend, anyway.
530  else {
531  //Compute SHA1 of payload
532  SHA1(uc_node->payload, uc_node->payload_len, payload_sha1);
533  payload_sha1_str = print_hex_string(payload_sha1, SHA_DIGEST_LENGTH);
534  //Make HexDump output out of binary payload
535  payload_hd_str = hex_dump(uc_node->payload, uc_node->payload_len, true); //must be freed
536  payload_str = print_hex_string(uc_node->payload, uc_node->payload_len); //must be freed
537 
538  json_do(false, ",\
539 \"payload_hd\":\"%s\", \
540 \"payload_str\":\"%s\", \
541 \"payload_sha1\":\"%s\"\
542 ",\
543  payload_hd_str,\
544  payload_str,\
