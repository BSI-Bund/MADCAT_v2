96 int init_pcap(char* dev, pcap_t **handle, const char* filter_exp)
97 {
98  char log_time[64] = ""; //Human readable start time (actual time zone)
99  char log_time_unix[64] = ""; //Unix timestamp (UTC)
100  struct timeval begin;
101  gettimeofday(&begin, NULL);
102  time_str(NULL, 0, log_time, sizeof(log_time)); //Get Human readable string only
103 
104  char errbuf[PCAP_ERRBUF_SIZE];// Error string
105  struct bpf_program fp; // The compiled filter
106  bpf_u_int32 mask; // Our netmask
107  bpf_u_int32 net; // Our IP
108 
109  fprintf(stderr, "%s [PID %d] Trying to set PCAP Filter Expression (fails if not root): \"%s\": ", log_time, getpid(), filter_exp);
110  fflush(stderr);
111 
112  // Find the properties for the device
113  if (pcap_lookupnet(dev, &net, &mask, errbuf) == -1)
114  return -1;
115  // Open the session in non-promiscuous mode
116  *handle = pcap_open_live(dev, BUFSIZ, 0, 100, errbuf);
117  if (handle == NULL)
118  return -2;
119  // Compile and apply the filter
120  if(strlen(filter_exp) > 0) {
121  if (pcap_compile(*handle, &fp, filter_exp, 0, net) == -1) {
122  fprintf(stderr, "ERROR compiling expression, ABORTING.\n");
123  fflush(stderr);
124  abort();
125  }
126  if (pcap_setfilter(*handle, &fp) == -1) {
127  fprintf(stderr, "ERROR setting expression, ABORTING.\n");
128  fflush(stderr);
129  abort();
130  }
131  free(fp.bf_insns);
132  }
133 
134  fprintf(stderr, "SUCCESS.\n");
135  fflush(stderr);
136 
137  return 0;
138 }
