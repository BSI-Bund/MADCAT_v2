45 
46  FILE *file = 0;
47  char* payload_hd_str = 0; //Payload as string in HexDump Format
48  char* payload_str = 0; //Payload as string
49  unsigned char payload_sha1[SHA_DIGEST_LENGTH]; //SHA1 of payload
50  char * payload_sha1_str = 0;
51  char file_name[2*PATH_LEN] = ""; //double path length for concatination purposes. PATH_LEN *MUST* be enforced when combinating path and filename!
52  struct timeval begin;
53  char log_time[64] = "";
54  char stop_time[64] = "";
55  char unix_time[64] = "";
56  char* hex_string = 0; //Hex string containing ICMP-Data
57  //Variables for inner packet analysis
58  bool tainted = false; //indicate errors while parsing
59  unsigned char* proto = 0; //protocol number of inner packet
60  int data_bytes = 0; //Bytes of data in inner packet (e.g. ICMP_UNREACH)
61  int data_offset = 0; //Data after end of 8-Byte ICMP-Header + data_offset, covering the parsed and JSONized data, is going to be dumped in a file.
62  //beginning time
63  long double unix_timeasdouble = time_str(unix_time, sizeof(unix_time), log_time, sizeof(log_time)); //...generate string with current time
64 
65  if (recv_len < 24) { //Minimum 20 Byte IP Header + 4 Byte ICMP Header. Should never happen.
66  fprintf(stderr, "%s ALERT: Paket to short for ICMP over IPv4, dumping %d Bytes of data:\n", log_time, recv_len);
67  print_hex(stderr, buffer, recv_len); //Dump malformed paket for analysis
68  return -1;
69  }
70  //Check IPv4 Header
71  ipv4icmp.ver = (uint8_t) (buffer[0] & 0b11110000) >> 4; //IPv4 should have set it's version field to, well, "4".
72  ipv4icmp.ihl = (uint8_t) ((buffer[0] & 0b00001111) * 32) / 8; //IP Header length is given in multipels of 32 bit or 4 Byte, respectivly
73  ipv4icmp.proto = (uint8_t) buffer[9]; //Proto should be 1 (ICMP), cause it's a RAW IP/ICMP Socket.
74  //Fetch IPs and convert them to strings.
75  ipv4icmp.src_ip = *(uint32_t*) (buffer+12);
76  ipv4icmp.src_ip_str = inttoa(ipv4icmp.src_ip);
77  ipv4icmp.dest_ip = *(uint32_t*) (buffer+16);
78  ipv4icmp.dest_ip_str = inttoa(ipv4icmp.dest_ip);
79  //Ignore Pakets, that have not been addressed to the IP given by the command line
80  if(strcmp(ipv4icmp.dest_ip_str, hostaddress) != 0 && strcmp("0.0.0.0", hostaddress) !=0) {
81  free(ipv4icmp.src_ip_str);
82  free(ipv4icmp.dest_ip_str);
83  return -1;
84  }
85  //Things that should never ever happen.
86  if( ipv4icmp.ver != 4 || ipv4icmp.ihl < 20 || ipv4icmp.ihl > 60 || (ipv4icmp.ihl + ICMP_HEADER_LEN) > recv_len || ipv4icmp.proto != 1 ) {
87  fprintf(stderr, "%s ALERT: Malformed Paket. Dumping %d Bytes of data:\n", log_time, recv_len);
88  print_hex(stderr, buffer, recv_len);
89  free(ipv4icmp.src_ip_str);
90  free(ipv4icmp.dest_ip_str);
91  return -1;
92  }

95  ipv4icmp.icmp_hdr = buffer + ipv4icmp.ihl;
96  //Fetch type and code by using the value from IP Header Length-Field,
97  // which has been check by the if statement above, so it should be save to use for addressing
98  ipv4icmp.type = *(uint8_t*) ipv4icmp.icmp_hdr;
99  ipv4icmp.code = *(uint8_t*) (ipv4icmp.icmp_hdr + sizeof(uint8_t));
100  ipv4icmp.icmp_check = ntohs(*(uint16_t*) (ipv4icmp.icmp_hdr + 2*sizeof(uint8_t)));
101  ipv4icmp.data = ipv4icmp.icmp_hdr + ICMP_HEADER_LEN;
102  ipv4icmp.data_len = recv_len - (ipv4icmp.ihl + ICMP_HEADER_LEN);
103  //Log connection
104  if(loglevel > 0) {
105  fprintf(stderr, "%s Received packet from %s to %s, type %u, code %u, with %ld Bytes of DATA.\n", log_time, \
106  ipv4icmp.src_ip_str, ipv4icmp.dest_ip_str, ipv4icmp.type, ipv4icmp.code, ipv4icmp.data_len);
107  } else {
108  fprintf(stderr, "%s Received packet from %s to %s, type %u, code %u, with %ld Bytes of DATA.\n", log_time, \
109  "<masked by default loglevel>", ipv4icmp.dest_ip_str, ipv4icmp.type, ipv4icmp.code, ipv4icmp.data_len);
110  }
111 
112  //Compute SHA1 of payload
113  SHA1(ipv4icmp.data, ipv4icmp.data_len, payload_sha1);
114  payload_sha1_str = print_hex_string(payload_sha1, SHA_DIGEST_LENGTH);
115  //Make HexDump output out of binary payload
116  payload_hd_str = hex_dump(ipv4icmp.data, ipv4icmp.data_len, true); //must be freed
117  payload_str = print_hex_string(ipv4icmp.data, ipv4icmp.data_len); //must be freed

120 {\
121 \"origin\":\"MADCAT\", \

143  Analyze whole inner packet.
144  a) UDP

146  c) default = ???

242  proto = ipv4icmp.data + 9;
243  //fprintf(stderr, "Inner Proto: %u\n", *proto); //Debug
244  switch(*proto) {

265  break;
