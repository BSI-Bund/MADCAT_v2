70  //Check IPv4 Header
71  ipv4udp.type = (uint8_t) (buffer[0] & 0b11110000) >> 4; //IPv4 should have set it's version field to, well, "4".
72  ipv4udp.ihl = (uint8_t) ((buffer[0] & 0b00001111) * 32) / 8; //IP Header length is given in multipels of 32 bit or 4 Byte, respectively
73  ipv4udp.proto = (uint8_t) buffer[9]; //Proto should be 17 (UDP), because it's a RAW IP/UDP Socket.
74  //Fetch IPs and convert them to strings.

76  ipv4udp.src_ip_str = inttoa(ipv4udp.src_ip);
77  ipv4udp.dest_ip = *(uint32_t*) (buffer+16);
78  ipv4udp.dest_ip_str = inttoa(ipv4udp.dest_ip);

82  fprintf(stderr, "%s ALERT: Malformed Paket. Dumping %d Bytes of data:\n", log_time, recv_len);
83  print_hex(stderr, buffer, recv_len);
84  free(ipv4udp.src_ip_str);
85  free(ipv4udp.dest_ip_str);
86  return -1;

92  ipv4udp.data = buffer + ipv4udp.ihl + UDP_HEADER_LEN;
