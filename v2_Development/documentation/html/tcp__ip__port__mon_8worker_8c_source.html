44 long int worker_tcp(char* dst_addr, int dest_port, char* src_addr, int src_port, long double timeout, \
45  char* data_path, int max_file_size, int s, FILE* confifo)
46 {
47  //on some systems, e.g. VMs, binding to a specific address does not work as expected.
48  if(strcmp(dst_addr, hostaddr) != 0 && strcmp("0.0.0.0", hostaddr) !=0) //char hostaddr[INET6_ADDRSTRLEN] globally defined.
49  return -1; //Filter packtes not matching hostaddress by returning from child
50 
51  int size_recv;
52  char chunk[CHUNK_SIZE];
53  char* payload = malloc(CHUNK_SIZE); //Paylaod (Binary)
54  char* payload_hd_str = 0; //Payload as string in HexDump Format
55  char* payload_str = 0; //Payload as string
56  unsigned char payload_sha1[SHA_DIGEST_LENGTH]; //SHA1 of payload
57  char * payload_sha1_str = 0;
58  long double timediff;
59  struct con_status_t con_status;
60  bool size_exceeded = false;
61  char log_time[64] = "";
62  char log_time_unix[64] = "";
63  long double unix_timeasdouble = time_str(log_time_unix, sizeof(log_time_unix), log_time, sizeof(log_time));
64  long double duration = 0;
65  long double min_rtt = 0;
66 
67  FILE *file = 0;
68  char file_name[2*PATH_LEN] = ""; //double path length for concatination purposes. PATH_LEN *MUST* be enforced when combinating path and filename!
69  char now_time[64] = "";
70  char lastrecv_time[64] = "";
71 
72  //structures for timeout measurment
73  struct timeval begin, now;
74 
75  //Log connection to STDERR in readeable format
76  if(loglevel>0) {
77  fprintf(stderr, "%s [PID %d] CONNECTION from %s:%d to %s:%d\n", log_time, getpid(), src_addr, src_port, dst_addr, dest_port);
78  } else {
79  fprintf(stderr, "%s [PID %d] CONNECTION from %s:%d to %s:%d\n", log_time, getpid(), "<Masked by default loglevel>", src_port, dst_addr, dest_port);
80  }
81 
82  //Begin new global JSON output and...
83  json_do(true, "");
84  //Log connection in json-format (Suricata-like).
85  json_do(false,"{\
86 \"src_ip\":\"%s\", \
87 \"dest_port\": %d, \
88 \"timestamp\":\"%s\", \
89 \"dest_ip\":\"%s\", \
90 \"src_port\": %d, \
91 \"proto\":\"TCP\", \
92 \"event_type\":\"flow\", \
93 \"unixtime\": %s, \
94 ",\
95  src_addr, \
96  dest_port, \
97  log_time, \
98  dst_addr, \
99  src_port, \
100  log_time_unix \
101  );
102 
103  //Generate connection tag to identify connection. Maximum is 28 Bytes, e.g. "123.456.789.012_43210_98765\0"
104  snprintf(con_status.tag, 28, "%s_%d_%d", src_addr, dest_port, src_port);
105  //initialize connection state for connection con_status by postprocessor
106  snprintf(con_status.state, 16, "%s", "open");
107  snprintf(con_status.reason, 16, "%s", "n/a");
108  snprintf(con_status.start, 64, "%s", log_time);
109  snprintf(con_status.end, 64, "%s", log_time);
110  con_status.timeasdouble = unix_timeasdouble;
111  con_status.data_bytes = 0;
112 
113  //make socket non blocking
114  fcntl(s, F_SETFL, O_NONBLOCK);
115 
116  //initialize beginning time and time now for first run
117  gettimeofday(&begin, NULL);
118  bool firstpacket = true;
119  while(1) { //receiving loop
120  //get current time
121  gettimeofday(&now, NULL); //now is the receiving time
122  time_str(NULL, 0, now_time, sizeof(now_time)); //Get Human readable string only
123  timediff = (now.tv_sec - begin.tv_sec) + 1e-6 * (now.tv_usec - begin.tv_usec); //time elapsed in seconds
124  //break after timeout
125  if(timediff > timeout) {
126  if (!size_exceeded) { //test if size has been exceeded (con_status.data_bytes >= max_file_size) to not overwritte con_status.reason.
127  snprintf(con_status.reason, 16, "%s", "timeout");
128  }
129  break;
130  }
131  //test if max_file_size is exceeded
132  if((con_status.data_bytes >= max_file_size) && max_file_size >= 0) {
133  snprintf(con_status.reason, 16, "%s", "size exceeded");
134  size_exceeded = true;
135  }
136 
137  memset(chunk,0, CHUNK_SIZE); //clear the variable
138  size_recv = recv(s, chunk, CHUNK_SIZE, 0);
139  if(size_recv <= 0) {
140  //if nothing was received then we want to wait a little before trying again, 0.1 seconds
141  usleep(50000);
142  } else {
143  //reset beginning time
144  gettimeofday(&begin, NULL);
145  con_status.data_bytes += size_recv; //calculate totale size received
146  if (con_status.data_bytes > 0 && !size_exceeded) { //proceed for writing payload in file / JSON only if max_file_size has not been exceeded.
147 
148  if (file == 0) { //if somthing had been received and no file is open yet...
149  //...generate filename LinuxTimeStamp-milisecends_destinationAddress-destinationPort_sourceAddress-sourcePort.tpm
150  sprintf(file_name, "%s%s_%s-%d_%s-%d.tpm", data_path, log_time, dst_addr, dest_port, src_addr, src_port);
151  file_name[PATH_LEN-1] = 0; //Enforcing PATH_LEN
152  if(loglevel>0) {
153  fprintf(stderr, "%s [PID %d] FILENAME: %s\n",log_time, getpid(), file_name);
154  } else {
155  fprintf(stderr, "%s [PID %d] FILENAME: %s%s_%s-%d_%s-%d.tpm\n",log_time, getpid(), \
156  data_path, log_time, dst_addr, dest_port, "<Masked by default loglevel>", src_port);
157  }
158  file = fopen(file_name,"wb"); //Open File
159  }
160  //Write when -and only WHEN nothing went wrong- data in chunk to file
161  if (file != 0) {
162  fwrite(chunk, size_recv, 1, file);
163  CHECK(fflush(file), == 0);
164  //Save Payload for JSON-Output
165  payload = realloc(payload, con_status.data_bytes); //get memory for all received bytes so far
166  memcpy(payload + con_status.data_bytes - size_recv, chunk, size_recv); //copy chunk to payload
167  } else { //if somthing went wrong, abort.
168  fprintf(stderr, "%s [PID %d] ERROR: Could not write to file %s\n",now_time, getpid(), file_name);
169  free(payload);
170  abort();
171  }
172  }
173  long double duration_saved = duration;
174  duration = time_str(NULL, 0, lastrecv_time, sizeof(lastrecv_time)) - con_status.timeasdouble;
175  if(!firstpacket && ( duration - duration_saved < min_rtt || min_rtt == 0) ) {
176  min_rtt = duration - duration_saved;
177  }
178  snprintf(con_status.end, sizeof(con_status.end), "%s", lastrecv_time); //save current time as end time candidate
179  firstpacket = false;
180  }
181  } //end of receiving loop
182  //if a file has been opened, because a stream had been received, close its filepointer to prevent data loss.
183  if (file != 0) {
184  fclose(file);
185  if(loglevel>0) {
186  fprintf(stderr, "%s [PID %d] FILE %s closed\n", now_time, getpid(), file_name);
187  } else {
188  fprintf(stderr, "%s [PID %d] FILE %s%s_%s-%d_%s-%d.tpm closed\n",log_time, getpid(), \
189  data_path, log_time, dst_addr, dest_port, "<Masked by default loglevel>", src_port);
190  }
191  }
192  snprintf(con_status.state, 16, "%s", "closed");
193 
194  //Compute SHA1 of payload
195  SHA1(payload, (size_exceeded ? max_file_size : con_status.data_bytes), payload_sha1);
196  payload_sha1_str = print_hex_string(payload_sha1, SHA_DIGEST_LENGTH);
197  //Make HexDump output out of binary payload
198  payload_hd_str = hex_dump(payload, (size_exceeded ? max_file_size : con_status.data_bytes), true); //must be freed
199  payload_str = print_hex_string(payload, (size_exceeded ? max_file_size : con_status.data_bytes)); //must be freed
200  //Log flow information in json-format (Suricata-like)
201  json_do(false,"\
202 \"FLOW\":{\
203 \"start\":\"%s\", \
204 \"end\":\"%s\", \
205 \"duration\":%Lf, \
206 \"min_rtt\":%Lf, \
207 \"state\":\"%s\", \
208 \"reason\":\"%s\", \
209 \"bytes_toserver\": %ld,\
210 \"payload_hd\":\"%s\",\
211 \"payload_str\":\"%s\",\
212 \"payload_sha1\":\"%s\"\
213 }}",\
214  con_status.start,\
215  con_status.end,\
216  duration,\
217  min_rtt,\
218  con_status.state,\
219  con_status.reason,\
220  con_status.data_bytes,\
221  payload_hd_str,\
222  payload_str,\
223  payload_sha1_str
224  );
225 
226 
227 #if DEBUG >= 2
228  int consem_val = -127;
229  CHECK(sem_getvalue(consem, &consem_val), != -1); //Ceck
230  fprintf(stderr, "*** DEBUG [PID %d] Acquire lock for output.\n", getpid());
231  fprintf(stderr, "%s [PID %d] : Value of connection semaphore: %d.\n", log_time, getpid(), consem_val);
232 #endif
233  struct timespec sem_timeout; //time to wait in sem_timedwait() call
234  clock_gettime(CLOCK_REALTIME, &sem_timeout);
235  sem_timeout.tv_sec += 1;
236  sem_timedwait(consem, &sem_timeout); //Acquire lock for output
237  char* tmp_str = json_do(false, "");
238  fprintf(confifo,"%s\n", tmp_str); //print json output for further analysis
239  fflush(confifo);
240 #if DEBUG >= 2
241  fprintf(stderr, "*** DEBUG [PID %d] Release lock for output\n", getpid());
242 #endif
243  sem_post(consem); //release lock
244 
245 
246  fprintf(stdout,"{\"CONNECTION\": %s}\n", json_do(false, "")); //print json output for logging
247  fflush(stdout);
248 
249  if(loglevel>0) {
250  fprintf(stderr, "%s [PID %d] END of connection from %s:%d started %s\n",now_time, getpid(), src_addr, src_port, log_time);
251  } else {
252  fprintf(stderr, "%s [PID %d] END of connection from %s:%d started %s\n",now_time, getpid(), "<Masked by default loglevel>", src_port, log_time);
253  }
254 
255 
256  free(payload_sha1_str);
257  free(payload_str);
258  free(payload_hd_str);
259  free(payload);
260  return con_status.data_bytes;
261 }
