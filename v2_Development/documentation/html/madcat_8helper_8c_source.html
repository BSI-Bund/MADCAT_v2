43 long double time_str(char* unix_buf, int unix_size, char* readable_buf, int readable_size)
44 {
45  struct timeval tv;
46  char tmbuf[readable_size];
47  char tmzone[6]; //e.g. "+0100\0" is max. 6 chars
48 
49  gettimeofday(&tv, NULL); //fetch struct timeval with actuall time and convert it to string...
50 
51  if (readable_buf == NULL && unix_buf == NULL) return (long double) tv.tv_sec + (long double) tv.tv_usec * 1e-6; //Return unixtime, if no pointer given
52 
53  strftime(tmbuf, readable_size, "%Y-%m-%dT%H:%M:%S", localtime(&tv.tv_sec)); //Target format: "2018-08-17T05:51:53.835934", therefore...
54  strftime(tmzone, 6, "%z", localtime(&tv.tv_sec)); //...get timezone...
55  //...and finally print time and ms to string, append timezone and ensure it is null terminated.
56  if (readable_buf != NULL) {
57  snprintf(readable_buf, readable_size, "%s.%06ld%s", tmbuf, tv.tv_usec, tmzone);
58  readable_buf[readable_size-1] = 0; //Human readable string
59  }
60  if (unix_buf != NULL) {
61  snprintf(unix_buf, unix_size, "%lu.%lu", tv.tv_sec, tv.tv_usec);
62  unix_buf[unix_size-1] = 0; //Unix time incl. usec
63  }
64  return (long double) tv.tv_sec + (long double) tv.tv_usec * 1e-6; //Return unixtime as double value
65 }

129 unsigned char* hex_dump(const void *addr, int len, const bool json)
130 {
131  int i =0;
132  unsigned char ascii_buff[17]; //size is 16 character + \0
133  const unsigned char *pc = (const unsigned char*)addr;
134  //Hex output is 3 characters per Byte e.g. "ff " for 16 Bytes per row plus offset, ascii and padding with spaces. Number of rows is len div 16 plus first row.
135  int out_len = (16 * 3 + 32) * (len / 16 + 1);
136  unsigned char* output = malloc(out_len); //must be freed
137  unsigned char* out_ptr = output;
138  memset(output, 0, out_len);
139 
140  if (len == 0) {
141  return output;
142  }
143  if (len < 0) {
144  return output;
145  }
146  //Cap length to prevent possible overflow in output.
147  //Okay. It's at 4GB...
148  if (len > 0xFFFFFFFF) {
149  len = 0xFFFFFFFF;
150  }
151 
152  // Process every byte in the data.
153  for (i = 0; i < len; i++) {
154  // Multiple of 16 means new line (with line offset).
155 
156  if ((i % 16) == 0) {
157  // Just don't print ASCII for the zeroth line.
158  if (i != 0) {
159  out_ptr += snprintf(out_ptr, out_len - (out_ptr - output)," |%s|", ascii_buff);
160 
161  if (json)
162  out_ptr += snprintf(out_ptr, out_len - (out_ptr - output),"\\n");
163  else
164  out_ptr += snprintf(out_ptr, out_len - (out_ptr - output),"\n");
165  }
166 
167  // Output the offset.
168  out_ptr += snprintf(out_ptr, out_len - (out_ptr - output),"%08x ", i);
169  } else if ((i % 8) == 0) {
170  if (i != 0)
171  out_ptr += snprintf(out_ptr, out_len - (out_ptr - output)," ");
172  }
173 
174 
175  // Now the hex code for the specific character.
176  out_ptr += snprintf(out_ptr, out_len - (out_ptr - output)," %02x", pc[i]);
177 
178  // And store a printable ASCII character for later.
179  if ((pc[i] < 0x20) || (pc[i] > 0x7e))
180  ascii_buff[i % 16] = '.';
181  else if (json && pc[i] == 0x22) //Do not insert " in JSON!
182  ascii_buff[i % 16] = '\'';
183  else if (json && pc[i] == 0x5c) //Do not insert \ in JSON!
184  ascii_buff[i % 16] = '/';
185  else
186  ascii_buff[i % 16] = pc[i];
187  ascii_buff[(i % 16) + 1] = '\0';
188  }
189 
190  // Pad out last line if not exactly 16 characters.
191  while ((i % 16) != 0) {
192  out_ptr += snprintf(out_ptr, out_len - (out_ptr - output)," ");
193  if ((i % 8) == 0)
194  out_ptr += snprintf(out_ptr, out_len - (out_ptr - output)," ");
195 
196  i++;
197  }
198 
199  // And print the final ASCII bit.
200  out_ptr += snprintf(out_ptr, out_len - (out_ptr - output)," |%s|", ascii_buff);
201  out_ptr = 0;
202 
203  return output;
204 }

215 char* json_do(bool reset, const char* format, ...)
216 {
217  static json_struct json; //static to hold data in json_struct after return from function
218  static bool first_run = true;
219  signed int numchars = 0; //number of chars to write
220  va_list valst; //variable argument list
221  va_start (valst, format);
222 
223  if (reset) { //should the json_struct be reseted?
224  if (!first_run && json.str != NULL) free(json.str);
225  first_run = false;
226  CHECK(json.str = malloc(1), != 0);
227  *json.str = 0; //add trailing \0 (empty string)
228  }
229 
230  //get number of chars to write
231  va_start (valst, format);
232  numchars = vsnprintf(NULL, 0, format, valst);
233  va_end(valst);
234 
235  //if an empty string has been provided as parameter, just return the pointer to actual string
236  if (numchars == 0) {
237  return json.str;
238  }
239 
240  //allocate new memory for chars to write
241  CHECK(json.str = realloc(json.str, strlen(json.str) + numchars + 1), != 0);
242 
243  //append chars to string
244  va_start(valst, format);
245  CHECK(vsnprintf(json.str + strlen(json.str), numchars + 1, format, valst), != 0);
246  va_end(valst);
247 
248  return json.str; //return pointer to (new) string
249 }
