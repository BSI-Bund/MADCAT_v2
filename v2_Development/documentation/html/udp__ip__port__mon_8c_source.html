46 void* cleanup_t()
47 {
48  pthread_setcancelstate(PTHREAD_CANCEL_ENABLE, NULL);
49  pthread_setcanceltype(PTHREAD_CANCEL_ASYNCHRONOUS, NULL);
50  while ( true ) {
51  //Scheduled Cleanup connections
52  sleep(pc->proxy_timeout + 1);
53  //fprintf(stderr, "*** Cleanup...\n");
54  uc_cleanup(uc, pc->proxy_timeout);
55 #if DEBUG >= 2
56  fprintf(stderr, "*** DEBUG List of aktive connections:\n");
57  uc_print_list(uc);
58 #endif
59  }
60  return NULL;
