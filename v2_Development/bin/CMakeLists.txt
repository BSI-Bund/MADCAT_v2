# build Executables

add_executable(icmp_mon
  icmp_mon.c
)

add_executable(tcp_ip_port_mon
  tcp_ip_port_mon.c
)

add_executable(udp_ip_port_mon
  udp_ip_port_mon.c
)

add_executable(raw_mon
  raw_mon.c
)

target_link_libraries(icmp_mon
  MadCatHelper
  IcmpMonCore
  ${PCAP_LIBRARY}
  OpenSSL::SSL
  Threads::Threads
  ${LUA_LIBRARY}
)
target_compile_options (icmp_mon PRIVATE ${GCC_FLAGS})

target_link_libraries(tcp_ip_port_mon
  MadCatHelper
  TcpIpPortMonCore
  TcpProxyCore
  ${LUA_LIBRARY}
  ${PCAP_LIBRARY}
  #${NFCT_LIBRARY} #enable for netfilter conntrack support in TCP Module and #define CT_ENABLED in tcp_ip_port_mon.common.h
  OpenSSL::SSL
  Threads::Threads
)
target_compile_options (tcp_ip_port_mon PRIVATE ${GCC_FLAGS})

target_link_libraries(udp_ip_port_mon
  MadCatHelper
  UdpIpPortMonCore
  ${PCAP_LIBRARY}
  OpenSSL::SSL
  Threads::Threads
  ${LUA_LIBRARY}
)
target_compile_options (udp_ip_port_mon PRIVATE ${GCC_FLAGS})

target_link_libraries(raw_mon
  RawMonCore
  ${PCAP_LIBRARY}
  OpenSSL::SSL
  Threads::Threads
  ${LUA_LIBRARY}
)
target_compile_options (raw_mon PRIVATE ${GCC_FLAGS})


install(TARGETS
          icmp_mon
          udp_ip_port_mon
          tcp_ip_port_mon
          raw_mon
        RUNTIME DESTINATION bin
        )
