97 int init_pcap(char* dev, char* dev_addr, pcap_t **handle, char* pcap_filter_str)
98 {
99  char errbuf[PCAP_ERRBUF_SIZE];// Error string
100  struct bpf_program fp; // The compiled ct_filter, global to get freed. Free here?
101  char filter_exp[ strlen(pcap_filter_str) + strlen(dev_addr) + 1 ]; //The ct_filter expression
102  bpf_u_int32 mask; // Our netmask
103  bpf_u_int32 net; // Our IP
104 
105  //Capture only TCP-SYN's...
106  strncpy(filter_exp, pcap_filter_str, sizeof(filter_exp));
107  //...for some systems (Linux Kernel >= 5 ???) own host IP has to be appended, so that the final ct_filter string looks like "tcp[tcpflags] & (tcp-syn) != 0 and tcp[tcpflags] & (tcp-ack) == 0 & dst host 1.2.3.4"
108  strncat(filter_exp, dev_addr, sizeof(filter_exp) - sizeof(pcap_filter_str));
109 
110 #if DEBUG >= 2
111  fprintf(stderr, "*** DEBUG [PID %d] PCAP Filter Expression: \"%s\"\n", getpid(), filter_exp);
112 #endif
113 
114  // Find the properties for the device
115  if (pcap_lookupnet(dev, &net, &mask, errbuf) == -1)
116  return -1;
117  // Open the session in non-promiscuous mode
118  *handle = pcap_open_live(dev, BUFSIZ, 0, 100, errbuf);
119  if (handle == NULL)
120  return -2;
121  // Compile and apply the ct_filter
122  if (pcap_compile(*handle, &fp, filter_exp, 0, net) == -1)
123  return -3;
124  if (pcap_setfilter(*handle, &fp) == -1)
125  return -4;
126 
127  free(fp.bf_insns);
128 
129  return 0;
130 }

168 //Signal Handler for parent watchdog
169 void sig_handler_parent(int signo)
170 {
171  static bool firstrun = true;
172  if (firstrun) {
173  char stop_time[64] = ""; //Human readable stop time (actual time zone)
174  time_str(NULL, 0, stop_time, sizeof(stop_time)); //Get Human readable string only
175  fprintf(stderr, "\n%s [PID %d] Parent Watchdog received Signal %s, shutting down...\n", stop_time, getpid(), strsignal(signo));
176 
177  int stat_accept = 0;
178 
179  for (int listenport = 1; listenport <65536; listenport++) {
180  if ( pc->portmap[listenport] && !waitpid(pctcp_get_lport(pc, listenport)->pid, &stat_accept, WNOHANG) )
181  kill(pctcp_get_lport(pc, listenport)->pid, SIGTERM);
182  }
183 
184  //Check if forked childs are still alive
185  //Give childs a chance to exit gracefull by sending SIGTERM
186  if ( !waitpid(listner_pid, &stat_accept, WNOHANG) )
187  kill(listner_pid, SIGTERM);
188 
189  if ( !waitpid(pcap_pid, &stat_accept, WNOHANG) )
190  kill(pcap_pid, SIGTERM);
191 
192  sleep(1); //Childs sometimes hanging while exit() or _exit() call, thus send SIGKILL
193  if ( !waitpid(listner_pid, &stat_accept, WNOHANG) )
194  kill(listner_pid, SIGKILL);
195  if ( !waitpid(pcap_pid, &stat_accept, WNOHANG) )
196  kill(pcap_pid, SIGKILL);
197 
198  sig_handler_common();
199  }
200  firstrun = true;
201  //exit parent process
202  exit(signo);
203  return;
204 }

322 int get_config_table(lua_State* L, char* name, struct proxy_conf_tcp_t* pc) //read proxy configuration from parsed LUA-File by luaL_dofile(...). Returns number of read elements.
323 {
324  char* backendaddr = 0;
325  int backendport = 0;
326 
327  int num_elements = 0;
328 
329  lua_getglobal(L, name); //push objekt "name" to stack and...
330 
331  if ( !lua_istable(L, -1) ) { //...check if this objekt is a table
332  fprintf(stderr, "\tNo proxy config found. Variable \"%s\" must be a LUA table.\n", name);
333  return num_elements;
334  }
335 
336  //Iterate over all possible portnumbers.
337  //Think about a more clever solution?
338  for (int listenport = 0; listenport<65536; listenport++) {
339 
340  lua_pushnumber(L, listenport); //push actuall portnumber on stack and...
341  lua_gettable(L, -2); //...call lua_gettable with this portnumber as key
342  if( !lua_isnil(L,-1) ) { //if corresponding value is not NIL...
343  lua_pushnumber(L, 1); //push "1" on the stack for the first elemnt in sub-table and...
344  lua_gettable(L, -2); //...fetch this entry
345  backendaddr = (char*) lua_tostring(L, -1);
346  lua_pop(L, 1); //remove result from stack
347 
348  lua_pushnumber(L, 2); //push "2" on the stack for the second elemnt in sub-table and...
349  lua_gettable(L, -2); //...fetch this entry
350  backendport = lua_tonumber(L, -1);
351  lua_pop(L, 1); //remove result from stack
352 
353  pctcp_push(pc, listenport, backendaddr, backendport);
354  pc->portmap[listenport] = true;
355  num_elements++;
356  }
357  lua_pop(L, 1); //remove sub-table from stack
358  }
359  return num_elements;
360 }
