80 void connection_really_close(struct epoll_event_handler* self)
81 {
82  struct connection_closure* closure = (struct connection_closure* ) self->closure;
83  struct data_buffer_entry* next;
84  while (closure->write_buffer != NULL) {
85  next = closure->write_buffer->next;
86  if (!closure->write_buffer->is_close_message) {
87  epoll_add_to_free_list(closure->write_buffer->data);
88  }
89  epoll_add_to_free_list(closure->write_buffer);
90  closure->write_buffer = next;
91  }
92 
93  epoll_remove_handler(self);
94  close(self->fd);
95  epoll_add_to_free_list(self->closure);
96  epoll_add_to_free_list(self);
97  rsp_log("Freed connection %p", self);
98 }

111 void connection_on_out_event(struct epoll_event_handler* self)
112 {
113  struct connection_closure* closure = (struct connection_closure*) self->closure;
114  int written;
115  int to_write;
116  struct data_buffer_entry* temp;
117  while (closure->write_buffer != NULL) {
118  if (closure->write_buffer->is_close_message) {
119  connection_really_close(self);
120  return;
121  }
122 
123  to_write = closure->write_buffer->len - closure->write_buffer->current_offset;
124  written = write(self->fd, closure->write_buffer->data + closure->write_buffer->current_offset, to_write);
125  if (written != to_write) {
126  if (written == -1) {
127  if (errno == ECONNRESET || errno == EPIPE) {
128  rsp_log_error("On out event write error");
129  connection_on_close_event(self);
130  return;
131  }
132  if (errno != EAGAIN && errno != EWOULDBLOCK) {
133  rsp_log_error("Error writing to client");
134  exit(-1);
135  }
136  written = 0;
137  }
138  closure->write_buffer->current_offset += written;
139  break;
140  } else {
141  temp = closure->write_buffer;
142  closure->write_buffer = closure->write_buffer->next;
143  epoll_add_to_free_list(temp->data);
144  epoll_add_to_free_list(temp);
145  }
146  }
147 }

150 void connection_on_in_event(struct epoll_event_handler* self)
151 {
152  struct connection_closure* closure = (struct connection_closure*) self->closure;
153  char read_buffer[BUFFER_SIZE];
154  int bytes_read;
155 
156  while ((bytes_read = read(self->fd, read_buffer, BUFFER_SIZE)) != -1 && bytes_read != 0) {
157  if (bytes_read == -1 && (errno == EAGAIN || errno == EWOULDBLOCK)) {
158  return;
159  }
160 
161  if (bytes_read == 0 || bytes_read == -1) {
162  connection_on_close_event(self);
163  return;
164  }
165 
166  if (closure->on_read != NULL) {
167  closure->on_read(closure->on_read_closure, read_buffer, bytes_read);
168  }
169  }

174 {

203 void connection_write(struct epoll_event_handler* self, char* data, int len)
204 {
205  struct connection_closure* closure = (struct connection_closure* ) self->closure;
206 
207  int written = 0;
208  if (closure->write_buffer == NULL) {
209  written = write(self->fd, data, len);
210  if (written == len) {
211  return;
212  }
213  }
214  if (written == -1) {
215  if (errno == ECONNRESET || errno == EPIPE) {
216  rsp_log_error("Connection write error");
217  connection_on_close_event(self);
218  return;
219  }
220  if (errno != EAGAIN && errno != EWOULDBLOCK) {
221  rsp_log_error("Error writing to client");
222  exit(-1);
223  }
224  written = 0;
225  }
226 
227  int unwritten = len - written;
228  struct data_buffer_entry* new_entry = malloc(sizeof(struct data_buffer_entry));
229  new_entry->is_close_message = 0;
230  new_entry->data = malloc(unwritten);
231  memcpy(new_entry->data, data + written, unwritten);
232  new_entry->current_offset = 0;
233  new_entry->len = unwritten;
234  new_entry->next = NULL;
235 
236  add_write_buffer_entry(closure, new_entry);
237 }

240 void connection_close(struct epoll_event_handler* self)
241 {
242  struct connection_closure* closure = (struct connection_closure* ) self->closure;
243  closure->on_read = NULL;
244  closure->on_close = NULL;
245  if (closure->write_buffer == NULL) {
246  connection_really_close(self);
247  } else {
248  struct data_buffer_entry* new_entry = malloc(sizeof(struct data_buffer_entry));
249  new_entry->is_close_message = 1;
250  new_entry->next = NULL;
251 
252  add_write_buffer_entry(closure, new_entry);
253  }
254 }
