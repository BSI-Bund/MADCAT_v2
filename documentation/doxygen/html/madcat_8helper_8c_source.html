53 long double time_str(char* unix_buf, int unix_size, char* readable_buf, int readable_size)
54 {
55  struct timeval tv;
56  char tmbuf[readable_size];
57  char tmzone[6]; //e.g. "+0100\0" is max. 6 chars
58 
59  gettimeofday(&tv, NULL); //fetch struct timeval with actuall time and convert it to string...
60 
61  if (readable_buf == NULL && unix_buf == NULL) return (long double) tv.tv_sec + (long double) tv.tv_usec * 1e-6; //Return unixtime, if no pointer given
62 
63  strftime(tmbuf, readable_size, "%Y-%m-%dT%H:%M:%S", localtime(&tv.tv_sec)); //Target format: "2018-08-17T05:51:53.835934", therefore...
64  strftime(tmzone, 6, "%z", localtime(&tv.tv_sec)); //...get timezone...
65  //...and finally print time and ms to string, append timezone and ensure it is null terminated.
66  if (readable_buf != NULL) {
67  snprintf(readable_buf, readable_size, "%s.%06ld%s", tmbuf, tv.tv_usec, tmzone);
68  readable_buf[readable_size-1] = 0; //Human readable string
69  }
70  if (unix_buf != NULL) {
71  snprintf(unix_buf, unix_size, "%lu.%lu", tv.tv_sec, tv.tv_usec);
72  unix_buf[unix_size-1] = 0; //Unix time incl. usec
73  }
74  return (long double) tv.tv_sec + (long double) tv.tv_usec * 1e-6; //Return unixtime as double value
75 }

138 char* hex_dump(const void *addr, int len, const bool json)
139 {
140  char* output = 0;
141 
142  if(len <= 0) { //return empty string
143  output = malloc(1);
144  memset(output, 0, 1);
145  return output;
146  }
147 
148  int i =0;
149  unsigned char ascii_buff[17]; //size is 16 character + \0
150  const unsigned char *pc = (const unsigned char*)addr;
151  //Hex output is 3 characters per Byte e.g. "ff " for 16 Bytes per row plus offset, ascii and padding with spaces. Number of rows is len div 16 plus first row.
152  int out_len = (16 * 3 + 32) * (len / 16 + 1);
153  output = malloc(out_len); //must be freed
154  char* out_ptr = output;
155  memset(output, 0, out_len);
156 
157  if (len == 0) {
158  return output;
159  }
160  if (len < 0) {
161  return output;
162  }
163  //Cap length to prevent possible overflow in output.
164  //Okay. It's at 4GB...
165  if (len > 0xFFFFFFFF) {
166  len = 0xFFFFFFFF;
167  }
168 
169  // Process every byte in the data.
170  for (i = 0; i < len; i++) {
171  // Multiple of 16 means new line (with line offset).
172 
173  if ((i % 16) == 0) {
174  // Just don't print ASCII for the zeroth line.
175  if (i != 0) {
176  out_ptr += snprintf(out_ptr, out_len - (out_ptr - output)," |%s|", ascii_buff);
177 
178  if (json)
179  out_ptr += snprintf(out_ptr, out_len - (out_ptr - output),"\\n");
180  else
181  out_ptr += snprintf(out_ptr, out_len - (out_ptr - output),"\n");
182  }
183 
184  // Output the offset.
185  out_ptr += snprintf(out_ptr, out_len - (out_ptr - output),"%08x ", i);
186  } else if ((i % 8) == 0) {
187  if (i != 0)
188  out_ptr += snprintf(out_ptr, out_len - (out_ptr - output)," ");
189  }
190 
191 
192  // Now the hex code for the specific character.
193  out_ptr += snprintf(out_ptr, out_len - (out_ptr - output)," %02x", pc[i]);
194 
195  // And store a printable ASCII character for later.
196  if ((pc[i] < 0x20) || (pc[i] > 0x7e))
197  ascii_buff[i % 16] = '.';
198  else if (json && pc[i] == 0x22) //Do not insert " in JSON!
199  ascii_buff[i % 16] = '\'';
200  else if (json && pc[i] == 0x5c) //Do not insert \ in JSON!
201  ascii_buff[i % 16] = '/';
202  else
203  ascii_buff[i % 16] = pc[i];
204  ascii_buff[(i % 16) + 1] = '\0';
205  }
206 
207  // Pad out last line if not exactly 16 characters.
208  while ((i % 16) != 0) {
209  out_ptr += snprintf(out_ptr, out_len - (out_ptr - output)," ");
210  if ((i % 8) == 0)
211  out_ptr += snprintf(out_ptr, out_len - (out_ptr - output)," ");
212 
213  i++;
214  }
215 
216  // And print the final ASCII bit.
217  out_ptr += snprintf(out_ptr, out_len - (out_ptr - output)," |%s|", ascii_buff);
218  out_ptr = 0;
219 
220  return output;
221 }

224 char* bitdump_ntoh(unsigned char* data, uint64_t len, char delimter, int block_len)
225 {
226  if(len < 0) return 0;
227 
228  long int bitstr_len = len*8 + (len*8-1)/block_len + 1; //length inculdes \0 termination
229  char* bitstr = malloc(bitstr_len);
230 
231  long int pos = 0;
232  uint16_t mask = 0b00000001;
233 
234  for(long int n = 0; n<len; n++)
235  {
236  #if defined(__LITTLE_ENDIAN_BITFIELD)
237  uint8_t byte = (data[n] & 0x0F) << 4 | (data[n] & 0xF0) >> 4;
238  byte = (byte & 0xCC) >> 2 | (byte & 0x33) << 2;
239  byte = (byte & 0xAA) >> 1 | (byte & 0x55) << 1;
240  #elif defined(__BIG_ENDIAN_BITFIELD)
241  #error "ERROR: Big Endian not supported yet"
242  #else
243  #error "Please fix <asm/byteorder.h>"
244  #endif
245 
246  for (long int i = 1; i<=8; i++)
247  {
248  if((pos+1) % (block_len+1) == 0)
249  {
250  bitstr[pos] = delimter;
251  //num_delimiter++;
252  pos++;
253  }
254 
255 
256  if(byte & mask)
257  {
258  bitstr[pos] = '1';
259  }
260  else
261  bitstr[pos] = '0';
262  byte = byte >> 1;
263 
264  //exp -= 1;
265  pos++;
266  }
267  }
268 
269  bitstr[pos] = 0;
270  return bitstr;
271 }

299 uint64_t pow64(uint64_t base, uint64_t exp)
300 {
301  uint64_t res = 1;
302  if(exp < 0) return 0;
303  for(uint64_t n = exp; n > 0; n--)
304  res *= base;
305  return res;
306 }
