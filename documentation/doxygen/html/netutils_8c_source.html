66 void make_socket_non_blocking(int socket_fd)
67 {
68  int flags;
69 
70  flags = fcntl(socket_fd, F_GETFL, 0);
71  if (flags == -1) {
72  rsp_log_error("Couldn't get socket flags");
73  exit(1);
74  }
75 
76  flags |= O_NONBLOCK;
77  if (fcntl(socket_fd, F_SETFL, flags) == -1) {
78  rsp_log_error("Couldn't set socket flags");
79  exit(-1);
80  }
81 }

84 int connect_to_backend(char* backend_host,
85  char* backend_port_str)
86 {
87  struct addrinfo hints;
88  memset(&hints, 0, sizeof(struct addrinfo));
89  hints.ai_family = AF_UNSPEC;
90  hints.ai_socktype = SOCK_STREAM;
91 
92  int getaddrinfo_error;
93  struct addrinfo* addrs;
94  getaddrinfo_error = getaddrinfo(backend_host, backend_port_str, &hints, &addrs);
95  if (getaddrinfo_error != 0) {
96  if (getaddrinfo_error == EAI_SYSTEM) {
97  rsp_log_error("Couldn't find backend (EAI_SYSTEM)");
98  } else {
99  rsp_log("Couldn't find backend: %s (%d)", gai_strerror(getaddrinfo_error), getaddrinfo_error);
100  }
101  exit(1);
102  }
103 
104  int backend_socket_fd;
105  struct addrinfo* addrs_iter;
106  for (addrs_iter = addrs;
107  addrs_iter != NULL;
108  addrs_iter = addrs_iter->ai_next) {
109  backend_socket_fd = socket(addrs_iter->ai_family,
110  addrs_iter->ai_socktype,
111  addrs_iter->ai_protocol);
112  if (backend_socket_fd == -1) {
113  continue;
114  }
115 
116  if (connect(backend_socket_fd,
117  addrs_iter->ai_addr,
118  addrs_iter->ai_addrlen) != -1) {
119  break;
120  }
121 
122 
123  close(backend_socket_fd);
124  }
125 
126  if (addrs_iter == NULL) {
127  rsp_log("Couldn't connect to backend");
128  exit(1);
129  }
130 
131  freeaddrinfo(addrs);
132 
133  return backend_socket_fd;
134 }
