41 //Helper function to parse IP Options. Returns tainted status, puts option data in hex string.
42 bool parse_ipopt(int opt_cpclno, const char* opt_name, \
43  unsigned char** opt_ptr_ptr, const unsigned char* beginofoptions_addr, const unsigned char* endofoptions_addr, struct dict** json)
44 {

48  struct dict* dict_temp = json_dict(false);
49  json = &dict_temp;
50  }

218  free(ip_daddr);

220 }

222 //returns amount of data bytes in buffer, regarding rec_len, NOT UDP length field, to prevent errors.
223 int analyze_udp_header(const unsigned char* packet, int recv_len, struct dict** json)
224 {
225  struct iphdr *iphdr = (struct iphdr *)(packet);
226 
227  int data_len = recv_len - (iphdr->ihl*4 + UDP_HEADER_LEN);
228  if(data_len <= 0) return -1; //Malformed Paket
229  unsigned char* data = (unsigned char*) packet + recv_len - data_len;
230  char* payload_sha1_str = NULL;
231  char* payload_hd_str = NULL;
232  char* payload_str = NULL;
233  unsigned char payload_sha1[SHA_DIGEST_LENGTH];
234 
235  struct udphdr *udphdr = (struct udphdr *) (packet + iphdr->ihl*4);
236 
237 
238  if(json == NULL) {
239  struct dict* dict_temp = json_dict(false);
240  json = &dict_temp;
241  }
242 
243  json_value.integer = ntohs(udphdr->source);
244  dict_update(*json, JSON_INT, json_value, 2, "UDP", "src_port");
245  json_value.integer = ntohs(udphdr->dest);
246  dict_update(*json, JSON_INT, json_value, 2, "UDP", "dest_port");
247  json_value.integer = ntohs(udphdr->len);
248  dict_update(*json, JSON_INT, json_value, 2, "UDP", "len");
249  json_value.integer = ntohs(udphdr->check);
250  dict_update(*json, JSON_INT, json_value, 2, "UDP", "checksum");
251 
252  json_value.integer = data_len;
253  dict_update(*json, JSON_INT, json_value, 2, "FLOW", "bytes_toserver");
254 
255  //Compute SHA1 of payload
256  SHA1(data, data_len, payload_sha1);
257  payload_sha1_str = print_hex_string(payload_sha1, SHA_DIGEST_LENGTH);
258  //Make HexDump output out of binary payload
259  payload_hd_str = hex_dump(data, data_len, true); //must be freed
260  payload_str = print_hex_string(data, data_len); //must be freed
261 
262  json_value.string = payload_hd_str;
263  dict_update(*json, JSON_STR, json_value, 2, "FLOW", "payload_hd");
264  json_value.string = payload_str;
265  dict_update(*json, JSON_STR, json_value, 2, "FLOW", "payload_str");
266  json_value.string = payload_sha1_str;
267  dict_update(*json, JSON_STR, json_value, 2, "FLOW", "payload_sha1");
268 
269  free(payload_hd_str);
270  free(payload_sha1_str);
271  free(payload_str);
272 
273  return data_len; //return number of data bytes after header in buffer
274 }
