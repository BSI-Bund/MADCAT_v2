42 //Capture only TCP-SYN's, for some sytems (Linux Kernel >= 5?) own host IPv4 or IPv6 has to be appended,
43 //thus the final filter string looks like "tcp[tcpflags] & (tcp-syn) != 0 and tcp[tcpflags] & (tcp-ack) == 0 & dst host 1.2.3.4"
44 #define PCAP_FILTER "tcp[tcpflags] & (tcp-syn) != 0 and tcp[tcpflags] & (tcp-ack) == 0 and dst host "
45 //Currently, TCPDump (and Wireshark) do not support embedded protocol BPF filters under IPv6
46 //Because IPv6 extension headers may be present, it is currently unsafe to detect TCP-Flags by using fixed offsets.
47 //Thus the final filter string looks like "ip6 and dst host 2021:1234:5678:9abc:def0::1"
48 //A filter string conataining "ip6 and tcp" does not wort, because the first extension header is evaluated by "tcp".
49 //Thus, if there are extensions headers present, and so the first one is not the Upper Layer Header containing TCP-Data, it will be discarded!
50 #define PCAP_FILTER_v6 "ip6 and dst host "
