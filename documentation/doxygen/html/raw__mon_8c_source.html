51  //Display Mascott and Version
52  if (argc == 2 && strcmp(argv[1], "version") == 0) {

60  //pseudo constant empty string e.g. for initialization of json_data_node_t and checks. Not used #define here, because this would lead to several instances of an empty constant string with different addresses.
61  EMPTY_STR[0] = 0;
62  //Start time
63  char log_time[64] = ""; //Human readable start time (actual time zone)
64  char log_time_unix[64] = ""; //Unix timestamp (UTC)

66  //Register Signal Handlers

71  //Parse command line.
72  char interface[64]= "";
73  int max_file_size = -1;
74  filter_exp = EMPTY_STR;
75 
76  // Checking if number of arguments is one (config file).
77  if (argc != 2) {
78  print_help_raw(argv[0]);
79  return -1;
80  }
81 
82  if (argc == 2) { //read config file
83  lua_State *luaState = lua_open();
84  if (luaL_dofile(luaState, argv[1]) != 0) {
85  fprintf(stderr, "%s [PID %d] Error parsing config file: %s\n\tRun without command line arguments for help.\n", log_time, getpid(), lua_tostring(luaState, -1));
86  exit(1);
87  }
88 
89  fprintf(stderr, "%s [PID %d] Parsing config file: %s\n", log_time, getpid(), argv[1]);
90 
91  strncpy(interface, get_config_opt(luaState, "interface"), sizeof(interface));
92  interface[sizeof(interface)-1] = 0; //copy interface and ensure null termination of this string. Ugly.
93  fprintf(stderr,"\tInterface: %s\n", interface);
94 
95  strncpy(user.name, get_config_opt(luaState, "user"), sizeof(user.name));
96  user.name[sizeof(user.name)-1] = 0;
97  fprintf(stderr, "\tuser: %s\n", user.name);
98 
99  long int filter_exp_len = strlen(get_config_opt(luaState, "raw_pcap_filter_exp"));
100  filter_exp = malloc(filter_exp_len + 1);
101  filter_exp[0] = 0;
102  strncpy(filter_exp, get_config_opt(luaState, "raw_pcap_filter_exp"), filter_exp_len + 1);
103  filter_exp[filter_exp_len] = 0;
104  fprintf(stderr, "\tPCAP Filter expression: %s\n", filter_exp);
105 
106  //check if mandatory string parameters are present, bufsize is NOT mandatory, the rest are numbers and are handled otherwise
107  if(strlen(interface) == 0 || strlen(user.name) == 0) {
108  fprintf(stderr, "%s [PID %d] Error in config file: %s\n", log_time, getpid(), argv[1]);
109  print_help_raw(argv[0]);
110  return -1;
111  }
112 
113  if(get_config_opt(luaState, "max_file_size") != EMPTY_STR) { //if optional parameter is given, set it.
114  max_file_size = atoi(get_config_opt(luaState, "max_file_size"));
115  }
116  fprintf(stderr, "\tmax_file_size: %d\n", max_file_size);
117 
118  if(get_config_opt(luaState, "loglevel") != EMPTY_STR) { //if optional parameter is given, set it.
119  loglevel = atoi(get_config_opt(luaState, "loglevel")); //convert string type to integer type (loglevel)
120  }
121  fprintf(stderr, "\tloglevel: %d\n", loglevel);
122 
123  fflush(stderr);
124  lua_close(luaState);
125  }
126 
127  fprintf(stderr, "%s [PID %d] Starting on interface %s\n", \
128  log_time, getpid(), interface);
129 
130  //Init pcap , drop priviliges, sniff for SYN-Packets and log them
131 
132 #if DEBUG >= 2
133  fprintf(stderr, "*** DEBUG [PID %d] Initialize PCAP\n", getpid());
134 #endif
135  CHECK(init_pcap(interface, &handle, filter_exp), == 0); //Init libpcap
136 
137  fprintf(stderr, "%s [PID %d] ", log_time, getpid());
138  drop_root_privs(user, "Sniffer"); //drop priviliges
139 
140  struct json_data_node_t json_data; //JSON Data structure for output generation
141  //Link (POINTERS!!!) timestamp(s) in JSON Data Structure to timestamp variables (multiple times to match the data model of other modules)
142 
143  char* payload_str = 0; //Payload as string
144  char* payload_hd_str = 0; //Payload as string in HexDump Format
145  char * payload_sha1_str = 0;
146  unsigned char payload_sha1[SHA_DIGEST_LENGTH]; //SHA1 of payload
147  bool size_exceeded = false; //max file size exceeded?
148  fprintf(stderr, "%s [PID %d] Sniffing...\n", log_time, getpid());
149  while (1) {
150  //Sniff packet
151  packet = 0;
152  packet = (unsigned char*) pcap_next(handle, &header); //Wait for and grab Packet (see PCAP_FILTER) (Maybe of maybe not BLOCKING!)
153 
154  //Test if something went wrong
155  if (packet == 0) continue;
156  if (!(header.len > ETHERNET_HEADER_LEN)) continue;
157 
158  //Preserve actuall time of Connection attempt, linked to timestamps in json_data
159  json_data.timeasdouble = time_str(log_time_unix, sizeof(log_time_unix), log_time, sizeof(log_time));
160  json_data.unixtime = log_time_unix;
161  json_data.timestamp = log_time;
162  json_data.start = log_time;
163  fprintf(stderr, "%s [PID %d] RAW Packet received\n", log_time, getpid());
164 
165  //Set pointer and length to address layer 3 directly in received data
166  int packet_len = header.len - ETHERNET_HEADER_LEN;
167  unsigned char* packet_layer3 = (unsigned char*) packet + ETHERNET_HEADER_LEN;
168 
169  if (packet_len > max_file_size && max_file_size > 0)
170  size_exceeded = true;
171  else
172  size_exceeded = false;
173 
174 #if DEBUG >= 2
175  fprintf(stdout, "\n%s\n", hex_dump(packet_layer3, packet_len, false));
176 #endif
177 
178  json_data.bytes_toserver = packet_len; //Len is defined here as layer 3 protokoll data + encapsulated protocols and their payload
179 
180  //Process packet data
181  //Compute SHA1 of packet
182  SHA1(packet_layer3, (size_exceeded ? max_file_size : packet_len), payload_sha1);
183  payload_sha1_str = print_hex_string(payload_sha1, SHA_DIGEST_LENGTH); //must be freed
184  //Make HexDump output out of binary packet contents
185  // entweder unsigned char als type in der Funktion oder payload_hd_str
186  payload_hd_str = hex_dump(packet_layer3, (size_exceeded ? max_file_size : packet_len), true); //must be freed
187  payload_str = print_hex_string(packet_layer3, (size_exceeded ? max_file_size : packet_len)); //must be freed
188 
189  //Begin new global JSON output and open JSON object
190  json_data.duration = time_str(NULL, 0, log_time, sizeof(log_time)) - json_data.timeasdouble;
191  json_data.end = log_time;
192 
193  json_value.string = "MADCAT";
194  dict_update(json_dict(true), JSON_STR, json_value, 1, "origin"); //begin new JSON Output
195  json_value.string = json_data.timestamp;
196  dict_update(json_dict(false), JSON_STR, json_value, 1, "timestamp");
197 
198  //Assumption of IP (v4 or v6) to determine version
199  struct ether_header * ethhdr = (struct ether_header *) packet; //Ethernet Header
200  uint16_t ether_type = ntohs(ethhdr->ether_type);
201  struct iphdr *iphdr = (struct iphdr *)(packet + ETHERNET_HEADER_LEN); //IPv4 header structure
202  struct ipv6hdr *ip6hdr = (struct ipv6hdr *) iphdr; //Interpretation of the header as IPv6 Header
203  char* proto_str = EMPTY_STR;
204  char* dest_ip = malloc(INET6_ADDRSTRLEN); dest_ip[0] = 0;
205  char* src_ip = malloc(INET6_ADDRSTRLEN); src_ip[0] = 0;
206  bool tainted = false;
207  //Source: https://www.iana.org/assignments/ieee-802-numbers/ieee-802-numbers.xhtml
208  switch(ether_type) { //Determine protocol type by ethertype
209  case 0x0800: json_data.proto = 4; break;
210  case 0x86DD: json_data.proto = 6; break;
211  default: json_data.proto = -1*ether_type; break;
212  }
213  //Check packet length for detected IP version.
214  //For IPv4 minimum ist 20bytes
215  //For IPv6 the Headerlength is fixed 40bytes
216  if ((json_data.proto == 4 && packet_len < IPV4_HEADER_MIN_LEN) ||
217  (json_data.proto == 6 && packet_len < IPV6_HEADER_LEN )) {
218  proto_str = malloc(10);
219  snprintf(proto_str, 10, "MALFORMED");
220  tainted = true;
221  break;
222  }
223  else {
224  //Try to determine IPv4/IPv6 transport protocol
225  switch(json_data.proto) {
226  case 4: //If IPv4 has been detected, no suffix is used (tcpdump-style)
227  proto_str = itoprotostr(iphdr->protocol, "");
228  inet_ntop(AF_INET, &(iphdr->saddr), src_ip, INET6_ADDRSTRLEN);
229  inet_ntop(AF_INET, &(iphdr->daddr), dest_ip, INET6_ADDRSTRLEN);
230  //Include IP Information only if IPv4/v6 has been detected
231  json_value.string = src_ip;
232  dict_update(json_dict(false), JSON_STR, json_value, 1, "src_ip");
233  json_value.string = dest_ip;
234  dict_update(json_dict(false), JSON_STR, json_value, 1, "dest_ip");
235  break;
236  case 6: //If IPv6 has been detected, the suffix "v6" is used
237  proto_str = itoprotostr(ip6hdr->nexthdr, "v6");
238  inet_ntop(AF_INET6, &(ip6hdr->saddr), src_ip, INET6_ADDRSTRLEN);
239  inet_ntop(AF_INET6, &(ip6hdr->daddr), dest_ip, INET6_ADDRSTRLEN);
240  //Include IP Information only if IPv4/v6 has been detected
241  json_value.string = src_ip;
242  dict_update(json_dict(false), JSON_STR, json_value, 1, "src_ip");
243  json_value.string = dest_ip;
244  dict_update(json_dict(false), JSON_STR, json_value, 1, "dest_ip");
245  break;
246  default: //If neither IPv4 nor IPv6 could be detected, raw ethertype is used
247  proto_str = malloc(20);
248  snprintf(proto_str, 20, "0x%04X", ether_type); break;
249  break;
250  }
251  }
252 
253  json_value.string = proto_str;
254  dict_update(json_dict(false), JSON_STR, json_value, 1, "proto");
255  json_value.string = "RAW";
256  dict_update(json_dict(false), JSON_STR, json_value, 1, "event_type");
257  json_value.boolean = tainted;
258  dict_update(json_dict(false), JSON_BOOL, json_value, 1, "tainted");
259  json_value.floating = atof(json_data.unixtime);
260  dict_update(json_dict(false), JSON_FLOAT, json_value, 1, "unixtime");
261  json_value.string = json_data.start;
262  dict_update(json_dict(false), JSON_STR, json_value, 2, "FLOW", "start");
263  json_value.string = json_data.end;
264  dict_update(json_dict(false), JSON_STR, json_value, 2, "FLOW", "end");
265  json_value.floating = json_data.duration;
266  dict_update(json_dict(false), JSON_FLOAT, json_value, 2, "FLOW", "duration");
267  json_value.string = "closed";
268  dict_update(json_dict(false), JSON_STR, json_value, 2, "FLOW", "state");
269  dict_update(json_dict(false), JSON_STR, json_value, 2, "FLOW", "reason");
270  json_value.integer = json_data.bytes_toserver;
271  dict_update(json_dict(false), JSON_INT, json_value, 2, "FLOW", "bytes_toserver");
272  json_value.string = payload_hd_str;
273  dict_update(json_dict(false), JSON_STR, json_value, 2, "FLOW", "payload_hd");
274  json_value.string = payload_str;
275  dict_update(json_dict(false), JSON_STR, json_value, 2, "FLOW", "payload_str");
276  json_value.string = payload_sha1_str;
277  dict_update(json_dict(false), JSON_STR, json_value, 2, "FLOW", "payload_sha1");
278  json_value.string = filter_exp;
279  dict_update(json_dict(false), JSON_STR, json_value, 2, "RAW", "pcap_filter");
280  json_value.hex.number = ether_type; json_value.hex.format = HEX_FORMAT_04;
281  dict_update(json_dict(false), JSON_HEX, json_value, 2, "RAW", "ether_type");
282 
283  //print JSON Object to stdout for logging
284  char* output = dict_dumpstr(json_dict(false));

288  }

298  }
299 
300  return 0;
