151 struct array* array_add(struct array* array, __uint8_t type, union json_type value) {
152  struct array* new = array;
153  if(array->next == 0) {
154  if(new->type != JSON_EMPTY) new = array_new();
155  array->next = new;
156  new->next = 0;
157  new->type = type;
158  switch(type) {
159  case JSON_NULL: break;
160  case JSON_BOOL: new->value.boolean = value.boolean; break;
161  case JSON_HEX:
162  new->value.hex.number = value.hex.number;
163  new->value.hex.format = value.hex.format;
164  break;
165  case JSON_INT: new->value.integer = value.integer; break;
166  case JSON_FLOAT: new->value.floating = value.floating; break;
167  case JSON_STR:
168  new->value.string = malloc(strlen(value.string)+1);
169  memset(new->value.string, 0, strlen(value.string)+1);
170  strncpy(new->value.string, value.string, strlen(value.string));
171  break;
172  case JSON_ARRAY: new->value.array = value.array; break;
173  case JSON_OBJ: new->value.object = value.object; break;
174  default: new->type = JSON_NULL; break;
175  }
176  new->next = 0;
177  return new;
178  } else {
179  array_add(array->next, type, value);
180  }
181  return NULL;
182 }

184 struct dict* dict_update(struct dict* dict, __uint8_t type, union json_type value, unsigned int path_len, ...) {
185  if(path_len < 1 || dict == 0) return NULL;
186  va_list valist;
187  char* next_key = 0;
188  struct dict* next_dict = dict;
189  struct dict* old_next_dict = 0;
190  struct dict* last_dict = 0;
191  struct dict* prev_dict = 0;
192 
193  //initialize valist
194  va_start(valist, path_len);
195  next_key = va_arg(valist, char *); //get first key of path
196 
197  if(path_len == 1) { //Trivial case with path_len == 1
198  //fprintf(stderr, "Trivial\n");
199  last_dict = __dict_plainsearch(dict, next_key, &prev_dict);
200  if(last_dict == 0) { //Element not existing, add new to dict directly
201  return __dict_add(dict, type, value, next_key); //adds new or updates empty
202  } else { //Element existing, empty first, than update
203  return __dict_add(dict_empty(last_dict), type, value, next_key); //adds new or updates empty
204  }
205  }
206 
207  //search given path till second last
208  for(int i = 0; i<path_len-1; i++) {
209  //fprintf(stderr, "i: %d\n", i);
210  //fprintf(stderr, "Following Path %s\n", next_key);
211  old_next_dict = next_dict;
212  if(i==0) //first dict has no need to descent
213  next_dict = __dict_plainsearch(next_dict, next_key, &prev_dict);
214  else //decent
215  next_dict = __dict_plainsearch(next_dict->value.object, next_key, &prev_dict);
216  //dict_printelement(stderr, next_dict);
217  if(next_dict == 0) { //Check if path is existing
218  #if DICT_DYN_PATH_UPDATES > 0
219  if(i==0) {
220  //next_dict = __dict_add(dict, JSON_OBJ, (union json_type) dict_new(), next_key); //if not make it existing
221  next_dict = dict_update(dict, JSON_OBJ, (union json_type) dict_new(), 1, next_key);
222  } else {
223  //next_dict = __dict_add(old_next_dict->value.object, JSON_OBJ, (union json_type) dict_new(), next_key); //if not make it existing
224  next_dict = dict_update(old_next_dict->value.object, JSON_OBJ, (union json_type) dict_new(), 1, next_key);
225  }
226 
227  if(old_next_dict->value.object == next_dict) //Check if parent dict is linked with last dict, if so it is the first one in a nested dict,...
228  next_dict->prev = old_next_dict; //...thus link it back to parent
229 
230  if(type == JSON_OBJ) //Link an inserted nested dict with parent dict
231  next_dict->value.object->prev = next_dict;
232 
233  #else
234  va_end(valist); //destroy valist
235  return NULL;
236  #endif
237  }
238  if(next_dict->type != JSON_OBJ){ //Check if a non-JSON Object is on path
239  //fprintf(stderr, "non-JSON Obj on path %s\n", next_key);
240  va_end(valist); //destroy valist
241  return NULL;
242  }
243  next_key = va_arg(valist, char *);
244  }
245 
246  va_end(valist); //destroy valist
247 
248  //get last key of path by descending in next_dict
249  last_dict = __dict_plainsearch(next_dict->value.object, next_key, &prev_dict);
250 
251  if(last_dict != 0){ //if update is exisiting in dict, make it empty first, thus possible included string(s), arrays or dicts must be freed
252  dict_empty(last_dict);
253  }
254 
255  //inserting *in* last nested dict in path
256  last_dict = __dict_add(next_dict->value.object, type, value, next_key);
257 
258  if(next_dict->value.object == last_dict) //Check if parent dict is linked with last dict, if so it is the first one in a nested dict,...
259  last_dict->prev = next_dict; //...thus link it back to parent
260 
261  if(type == JSON_OBJ) //Link an inserted nested dict with parent dict
262  last_dict->value.object->prev = last_dict;
263 
264  return last_dict;
265 }

267 struct dict* __dict_add(struct dict* dict, __uint8_t type, union json_type value, char* key) {
268  struct dict* new = dict;
269  if(dict->next == 0 || dict->type == JSON_EMPTY) { //Last element in list or Empty Element
270  //fprintf(stderr, "INSERT POINT FOUND!\n");
271  if(dict->type != JSON_EMPTY) { //Implicit ... && dict->next == 0, thus end of linked list and new dict-element needed.
272  new = dict_new();
273  new->next = 0;
274  new->prev = dict;
275  dict->next = new;
276  }
277  new->key = malloc(strlen(key)+1); //+1 for \0
278  memset(new->key, 0, strlen(key)+1);
279  strncpy(new->key, key, strlen(key));
280  new->type = type;
281  switch(type) {
282  case JSON_NULL: break;
283  case JSON_BOOL: new->value.boolean = value.boolean; break;
284  case JSON_HEX:
285  new->value.hex.number = value.hex.number;
286  new->value.hex.format = value.hex.format;
287  break;
288  case JSON_INT: new->value.integer = value.integer; break;
289  case JSON_FLOAT: new->value.floating = value.floating; break;
290  case JSON_STR:
291  new->value.string = malloc(strlen(value.string)+1);
292  memset(new->value.string, 0, strlen(value.string)+1);
293  strncpy(new->value.string, value.string, strlen(value.string));
294  break;
295  case JSON_ARRAY: new->value.array = value.array; break;
296  case JSON_OBJ: new->value.object = value.object; break;
297  default: new->type = JSON_NULL; break;
298  }
299  return new;
300  } else {
301  new = __dict_add(dict->next, type, value, key);
302  return new;
303  }
304  return new;
305 }

440 struct dict* dict_get(struct dict* dict, int path_len, ...) {
441  //fprintf(stderr,"DICT_GET with dict=%p, path_len=%d\n", dict, path_len);
442  va_list valist;
443  char* key = 0;
444  //initialize valist
445  va_start(valist, path_len);
446  if(dict == NULL) {
447  va_end(valist); //destroy valist
448  return NULL;
449  }
450  key = va_arg(valist, char *);
451  while(path_len > 0 && dict != NULL) {
452  //fprintf(stderr, "Searching for key %s...", key);
453  if(dict->key != NULL && strcmp(key, dict->key) == 0) { //found actuall part of path
454  if(path_len == 1) { //Object found
455  //fprintf(stderr, "Object found!\n");
456  return dict;
457  } else if(dict->type == JSON_OBJ) { //Point to descent found
458  dict = dict->value.object;
459  key = va_arg(valist, char *);
460  path_len--;
461  }
462  } else { //Iterate through list, till key ist found
463  dict = dict->next;
464  }
465  }
466  //destroy valist
467  va_end(valist);
468  //if(dict == NULL) fprintf(stderr,"Object *NOT* found!\n");
469  return dict;
470 }

531 struct dict* __dict_delrec(struct dict* dict, int path_len, char* next_key, va_list keys) {
532  if(dict == 0 || dict->key == 0) return NULL;
533 
534  //fprintf(stderr, "__dict_delrec: path_len %d next_key: %s\n", path_len, next_key);
535

537 
538  bool found = strcmp(next_key, dict->key) == 0 ? true : false;
539 
540  if(path_len == 1 && found) {
541  //fprintf(stderr, "DELETE: type: %d key: %s dict->next: %p\n", dict->type, dict->key, dict->next);
542  switch(dict->type) { //update exisiting in dict, handle appropriate
543  case JSON_OBJ:
544  dict_free(dict->value.object);
545  break;
546  case JSON_ARRAY:
547  array_free(dict->value.array);
548  break;
549  case JSON_STR:
550  free(dict->value.string);
551  break;
552  default:
553  break;

556  //fprintf(stderr, "dict->prev %p\n", dict->prev);
557  if(dict->prev) {

559  //fprintf(stderr, "Is nested!\n");
560  if(!dict->next){ //check if dict is also last element in the nested object
561  dict->prev->value.object = dict_new(); //if so, make new empty dict, so parent element is not left without (empty) content
562  } else
563  dict->prev->value.object = dict->next;
564  } else {
565  //fprintf(stderr, "Is *NOT* nested!\n");
566  dict->prev->next = dict->next;
567  }
568  }
569  if(dict->next) {
