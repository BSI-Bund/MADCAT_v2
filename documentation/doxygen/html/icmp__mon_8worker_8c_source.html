42 int worker_icmp(unsigned char* buffer, int recv_len, char* hostaddress, char* data_path)
43 {
44  struct ipv4icmp_t ipv4icmp; //struct to save IP-Header contents of intrest
45 
46  FILE *file = 0;
47  char* payload_hd_str = 0; //Payload as string in HexDump Format
48  char* payload_str = 0; //Payload as string
49  unsigned char payload_sha1[SHA_DIGEST_LENGTH]; //SHA1 of payload
50  char * payload_sha1_str = 0;
51  char file_name[2*PATH_LEN] = ""; //double path length for concatination purposes. PATH_LEN *MUST* be enforced when combinating path and filename!
52  char log_time[64] = "";
53  char stop_time[64] = "";
54  char unix_time[64] = "";
55  char* hex_string = 0; //Hex string containing ICMP-Data
56  //Variables for inner packet analysis
57  bool tainted = false; //indicate errors while parsing
58  unsigned char* proto = 0; //protocol number of inner packet
59  int data_bytes = 0; //Bytes of data in inner packet (e.g. ICMP_UNREACH)
60  int data_offset = 0; //Data after end of 8-Byte ICMP-Header + data_offset, covering the parsed and JSONized data, is going to be dumped in a file.
61  //beginning time
62  long double unix_timeasdouble = time_str(unix_time, sizeof(unix_time), log_time, sizeof(log_time)); //...generate string with current time
63 
64  if (recv_len < 24) { //Minimum 20 Byte IP Header + 4 Byte ICMP Header. Should never happen.
65  fprintf(stderr, "%s ALERT: Paket to short for ICMP over IPv4, dumping %d Bytes of data:\n", log_time, recv_len);
66  print_hex(stderr, buffer, recv_len); //Dump malformed paket for analysis
67  return -1;
68  }
69  //Check IPv4 Header
70  ipv4icmp.ver = (uint8_t) (buffer[0] & 0b11110000) >> 4; //IPv4 should have set it's version field to, well, "4".
71  ipv4icmp.ihl = (uint8_t) ((buffer[0] & 0b00001111) * 32) / 8; //IP Header length is given in multipels of 32 bit or 4 Byte, respectivly
72  ipv4icmp.proto = (uint8_t) buffer[9]; //Proto should be 1 (ICMP), cause it's a RAW IP/ICMP Socket.
73  //Fetch IPs and convert them to strings.
74  ipv4icmp.src_ip = *(uint32_t*) (buffer+12);
75  ipv4icmp.src_ip_str = inttoa(ipv4icmp.src_ip);
76  ipv4icmp.dest_ip = *(uint32_t*) (buffer+16);
77  ipv4icmp.dest_ip_str = inttoa(ipv4icmp.dest_ip);
78  //Ignore Pakets, that have not been addressed to the IP given by the command line
79  if(strcmp(ipv4icmp.dest_ip_str, hostaddress) != 0 && strcmp("0.0.0.0", hostaddress) !=0) {
80  free(ipv4icmp.src_ip_str);
81  free(ipv4icmp.dest_ip_str);
82  return -1;
83  }
84  //Things that should never ever happen.
85  if( ipv4icmp.ver != 4 || ipv4icmp.ihl < 20 || ipv4icmp.ihl > 60 || (ipv4icmp.ihl + ICMP_HEADER_LEN) > recv_len || ipv4icmp.proto != 1 ) {
86  fprintf(stderr, "%s ALERT: Malformed Paket. Dumping %d Bytes of data:\n", log_time, recv_len);
87  print_hex(stderr, buffer, recv_len);
88  free(ipv4icmp.src_ip_str);
89  free(ipv4icmp.dest_ip_str);
90  return -1;
91  }
92 
93  // ...and Parse ICMP-Header
94  ipv4icmp.icmp_hdr = buffer + ipv4icmp.ihl;
95  //Fetch type and code by using the value from IP Header Length-Field,
96  // which has been check by the if statement above, so it should be save to use for addressing
97  ipv4icmp.type = *(uint8_t*) ipv4icmp.icmp_hdr;
98  ipv4icmp.code = *(uint8_t*) (ipv4icmp.icmp_hdr + sizeof(uint8_t));
99  ipv4icmp.icmp_check = ntohs(*(uint16_t*) (ipv4icmp.icmp_hdr + 2*sizeof(uint8_t)));
100  ipv4icmp.data = ipv4icmp.icmp_hdr + ICMP_HEADER_LEN;
101  ipv4icmp.data_len = recv_len - (ipv4icmp.ihl + ICMP_HEADER_LEN);
102  //Log connection
103  if(loglevel > 0) {
104  fprintf(stderr, "%s Received packet from %s to %s, type %u, code %u, with %ld Bytes of DATA.\n", log_time, \
105  ipv4icmp.src_ip_str, ipv4icmp.dest_ip_str, ipv4icmp.type, ipv4icmp.code, ipv4icmp.data_len);
106  } else {
107  fprintf(stderr, "%s Received packet from %s to %s, type %u, code %u, with %ld Bytes of DATA.\n", log_time, \
108  "<masked by default loglevel>", ipv4icmp.dest_ip_str, ipv4icmp.type, ipv4icmp.code, ipv4icmp.data_len);
109  }
110 
111 
112 
113  //Open new JSON and log connection to STDOUT in json-format (Suricata-alike)
114  json_value.string = "MADCAT";
115  dict_update(json_dict(true), JSON_STR, json_value, 1, "origin");
116  json_value.string = log_time;
117  dict_update(json_dict(false), JSON_STR, json_value, 1, "timestamp");
118  json_value.floating = atof(unix_time);
119  dict_update(json_dict(false), JSON_FLOAT, json_value, 1, "unixtime");
120  json_value.string = ipv4icmp.src_ip_str;
121  dict_update(json_dict(false), JSON_STR, json_value, 1, "src_ip");
122  json_value.string = ipv4icmp.dest_ip_str;
123  dict_update(json_dict(false), JSON_STR, json_value, 1, "dest_ip");
124 
125  //Move to [ICMP][type] / [ICMP][code]?
126  json_value.integer = ipv4icmp.type;
127  dict_update(json_dict(false), JSON_INT, json_value, 1, "icmp_type");
128  json_value.integer = ipv4icmp.code;
129  dict_update(json_dict(false), JSON_INT, json_value, 1, "icmp_code");
130 
131  json_value.string = "ICMP";
132  dict_update(json_dict(false), JSON_STR, json_value, 1, "proto");
133  json_value.string = "flow";
134  dict_update(json_dict(false), JSON_STR, json_value, 1, "event_type");
135 
136  //Analyze Headers in ICMP-Payload
137  /******************************************
138  1. Analyze IP Header
139  2. Analyze ICMP Header in detail
140  A. Switch-case for types of intrest
141  i) if inner packet ist present (ICMP_UNREACH):
142  Analyze whole inner packet.
143  a) UDP
144  b) TCP
145  c) default = ???
146  4. Dump rest of packet
147  ******************************************/
148 
149  //Analyze IP Header
150  analyze_ip_header(buffer, recv_len, NULL);
151  //Analyze ICMP Header
152  json_value.integer = ipv4icmp.type;
153  dict_update(json_dict(false), JSON_INT, json_value, 2, "ICMP", "type");
154  json_value.integer = ipv4icmp.code;
155  dict_update(json_dict(false), JSON_INT, json_value, 2, "ICMP", "code");
156  json_value.hex.number = ipv4icmp.icmp_check;
157  json_value.hex.format = HEX_FORMAT_04;
158  dict_update(json_dict(false), JSON_HEX, json_value, 2, "ICMP", "checksum");
159 
160  switch(ipv4icmp.type) {
161  case MY_ICMP_ECHOREPLY: //print type_str, identifier and sequence
162  json_value.string = "echoreply";
163  dict_update(json_dict(false), JSON_STR, json_value, 2, "ICMP", "type_str");
164  json_value.hex.number = ntohs(*(uint16_t*) (ipv4icmp.icmp_hdr + 2*sizeof(uint16_t)));
165  json_value.hex.format = HEX_FORMAT_04;
166  dict_update(json_dict(false), JSON_HEX, json_value, 2, "ICMP", "id");
167  json_value.integer = ntohs(*(uint16_t*) (ipv4icmp.icmp_hdr + 3*sizeof(uint16_t)));
168  dict_update(json_dict(false), JSON_HEX, json_value, 2, "ICMP", "seq");
169  break;
170  case MY_ICMP_ECHO: //print type_str, identifier and sequence
171  json_value.string = "echo";
172  dict_update(json_dict(false), JSON_STR, json_value, 2, "ICMP", "type_str");
173  json_value.hex.number = ntohs(*(uint16_t*) (ipv4icmp.icmp_hdr + 2*sizeof(uint16_t)));
174  json_value.hex.format = HEX_FORMAT_04;
175  dict_update(json_dict(false), JSON_HEX, json_value, 2, "ICMP", "id");
176  json_value.integer = ntohs(*(uint16_t*) (ipv4icmp.icmp_hdr + 3*sizeof(uint16_t)));
177  dict_update(json_dict(false), JSON_HEX, json_value, 2, "ICMP", "seq");
178  break;
179  case MY_ICMP_UNREACH:
180  json_value.string = "unreach";
181  dict_update(json_dict(false), JSON_STR, json_value, 2, "ICMP", "type_str");
182  json_value.hex.number = *(uint32_t*) (ipv4icmp.icmp_hdr + 2*sizeof(uint16_t));
183  json_value.hex.format = HEX_FORMAT_08;
184  dict_update(json_dict(false), JSON_HEX, json_value, 2, "ICMP", "unused");
185 
186  switch(ipv4icmp.code) {
187  case MY_ICMP_NET_UNREACH:
188  json_value.string = "net_unreach";
189  dict_update(json_dict(false), JSON_STR, json_value, 2, "ICMP", "code_str");
190  break;
191  case MY_ICMP_HOST_UNREACH:
192  json_value.string = "host_unreach";
193  dict_update(json_dict(false), JSON_STR, json_value, 2, "ICMP", "code_str");
194  break;
195  case MY_ICMP_PROT_UNREACH:
196  json_value.string = "prot_unreach";
197  dict_update(json_dict(false), JSON_STR, json_value, 2, "ICMP", "code_str");
198  break;
199  case MY_ICMP_PORT_UNREACH:
200  json_value.string = "port_unreach";
201  dict_update(json_dict(false), JSON_STR, json_value, 2, "ICMP", "code_str");
202  break;
203  case MY_ICMP_FRAG_NEEDED:
204  json_value.string = "frag_needed";
205  dict_update(json_dict(false), JSON_STR, json_value, 2, "ICMP", "code_str");
206  break;
207  case MY_ICMP_SR_FAILED:
208  json_value.string = "sr_failed";
209  dict_update(json_dict(false), JSON_STR, json_value, 2, "ICMP", "code_str");
210  break;
211  case MY_ICMP_NET_UNKNOWN:
212  json_value.string = "net_unknown";
213  dict_update(json_dict(false), JSON_STR, json_value, 2, "ICMP", "code_str");
214  break;
215  case MY_ICMP_HOST_UNKNOWN:
216  json_value.string = "host_unknown";
217  dict_update(json_dict(false), JSON_STR, json_value, 2, "ICMP", "code_str");
218  break;
219  case MY_ICMP_HOST_ISOLATED:
220  json_value.string = "host_isolated";
221  dict_update(json_dict(false), JSON_STR, json_value, 2, "ICMP", "code_str");
222  break;
223  case MY_ICMP_NET_ANO:
224  json_value.string = "net_ano";
225  dict_update(json_dict(false), JSON_STR, json_value, 2, "ICMP", "code_str");
226  break;
227  case MY_ICMP_HOST_ANO:
228  json_value.string = "host_ano";
229  dict_update(json_dict(false), JSON_STR, json_value, 2, "ICMP", "code_str");
230  break;
231  case MY_ICMP_NET_UNR_TOS:
232  json_value.string = "net_unr_tos";
233  dict_update(json_dict(false), JSON_STR, json_value, 2, "ICMP", "code_str");
234  break;
235  case MY_ICMP_HOST_UNR_TOS:
236  json_value.string = "host_unr_tos";
237  dict_update(json_dict(false), JSON_STR, json_value, 2, "ICMP", "code_str");
238  break;
239  case MY_ICMP_PKT_FILTERED:
240  json_value.string = "pkt_filtered";
241  dict_update(json_dict(false), JSON_STR, json_value, 2, "ICMP", "code_str");
242  break;
243  case MY_ICMP_PREC_VIOLATION:
244  json_value.string = "prec_vioalation";
245  dict_update(json_dict(false), JSON_STR, json_value, 2, "ICMP", "code_str");
246  break;
247  case MY_ICMP_PREC_CUTOFF:
248  json_value.string = "prec_cutoff";
249  dict_update(json_dict(false), JSON_STR, json_value, 2, "ICMP", "code_str");
250  break;
251  default:
252  json_value.string = "tainted/unkown";
253  dict_update(json_dict(false), JSON_STR, json_value, 2, "ICMP", "code_str");
254  tainted = true;
255  break;
256  } //End of switch(ipv4icmp.code)
257  //Analyze inner IP-Header
258  struct dict* json_unreach = dict_new();
259  tainted = analyze_ip_header(ipv4icmp.data, recv_len, &json_unreach);
260  if(tainted) { //if inner IP-Header is tainted (e.g. < 20Bytes), also set tainted = true and break
261  if(!dict_append(json_unreach, dict_get(json_dict(false), 1, "ICMP")->value.object))
262  dict_free(json_unreach);
263  tainted = true;
264  break;
265  }
266  //10th Byte (count begins at 0, so it's data+9) contains the protocol number of inner IP-Header
267  //It has been parsed by above analyze_ip_headeripv4icmp.data + 4, recv_len) and therefore checked, so this access should be okay.
268  proto = ipv4icmp.data + 9;
269  switch(*proto) {
270  case 6: //TCP: data_offset is the whole length of inner packet minus number of data bytes after the IP/TCPs headers.
271  data_bytes = analyze_tcp_header(ipv4icmp.data, ipv4icmp.data_len, &json_unreach);
272  if(data_bytes < 0) {
273  tainted = true;
274  break;
275  }
276  data_offset = ipv4icmp.data_len - data_bytes;
277  break;
278  case 17: //UDP: data_offset is the whole length of inner packet minus number of data bytes after the IP/UDP headers.
279  data_bytes = analyze_udp_header(ipv4icmp.data, ipv4icmp.data_len, &json_unreach);
280  if(data_bytes < 0) {
281  tainted = true;
282  break;
283  }
284  data_offset = ipv4icmp.data_len - data_bytes;
285  break;
286  case 1: //TODO: ICMP in ICMP
287  default: //protocol unknown or tainted
288  tainted = true;
289  break;
290  }
291  json_value.object = json_unreach;
292  if(!dict_append(json_unreach, dict_get(json_dict(false), 1, "ICMP")->value.object))
293  dict_free(json_unreach);
294  break;
295  case MY_ICMP_SOURCEQUENCH:
296  json_value.string = "sourcequench";
297  dict_update(json_dict(false), JSON_STR, json_value, 2, "ICMP", "type_str");
298  break;
299  case MY_ICMP_REDIRECT:
300  json_value.string = "redirect";
301  dict_update(json_dict(false), JSON_STR, json_value, 2, "ICMP", "type_str");
302  break;
303  case MY_ICMP_ALTHOST:
304  json_value.string = "althost";
305  dict_update(json_dict(false), JSON_STR, json_value, 2, "ICMP", "type_str");
306  break;
307  case MY_ICMP_RTRADVERT:
308  json_value.string = "rtradvert";
309  dict_update(json_dict(false), JSON_STR, json_value, 2, "ICMP", "type_str");
310  break;
311  case MY_ICMP_RTRSOLICIT:
312  json_value.string = "rtrsolicit";
313  dict_update(json_dict(false), JSON_STR, json_value, 2, "ICMP", "type_str");
314  break;
315  case MY_ICMP_TIMXCEED:
316  json_value.string = "timxceed";
317  dict_update(json_dict(false), JSON_STR, json_value, 2, "ICMP", "type_str");
318  break;
319  case MY_ICMP_PARAMPROB:
320  json_value.string = "paramprob";
321  dict_update(json_dict(false), JSON_STR, json_value, 2, "ICMP", "type_str");
322  break;
323  case MY_ICMP_TSTAMP:
324  json_value.string = "tstamp";
325  dict_update(json_dict(false), JSON_STR, json_value, 2, "ICMP", "type_str");
326  break;
327  case MY_ICMP_TSTAMPREPLY:
328  json_value.string = "tstampreply";
329  dict_update(json_dict(false), JSON_STR, json_value, 2, "ICMP", "type_str");
330  break;
331  case MY_ICMP_IREQ:
332  json_value.string = "ireq";
333  dict_update(json_dict(false), JSON_STR, json_value, 2, "ICMP", "type_str");
334  break;
335  case MY_ICMP_IREQREPLY:
336  json_value.string = "ireqreply";
337  dict_update(json_dict(false), JSON_STR, json_value, 2, "ICMP", "type_str");
338  break;
339  case MY_ICMP_MASKREQ:
340  json_value.string = "maskreq";
341  dict_update(json_dict(false), JSON_STR, json_value, 2, "ICMP", "type_str");

361  dict_update(json_dict(false), JSON_STR, json_value, 2, "ICMP", "type_str");

363  break;
364  } //End of switch(ipv4icmp.type)
365  //if some data has been received (payload or tainted), that has not been parsed into JSON object yet, save the rest of datagram in a file
366  // e.g. TCP or UDP data in ICPM_UNREACH or data at the end of an ICMP Echo-Request/-Reply
367  // Also dump all data, if packet is marked as tainted
368  if(ipv4icmp.data_len - data_offset > 0 || tainted) { //payload data is left or tainted
369  //Generate filename LinuxTimeStamp-milisecends_destinationAddress-destinationPort_sourceAddress-sourcePort.tpm
370  sprintf(file_name, "%s%s_%s_%s-%u_%u.ipm", data_path, log_time, ipv4icmp.dest_ip_str, ipv4icmp.src_ip_str, ipv4icmp.type, ipv4icmp.code);
371  file_name[PATH_LEN-1] = 0; //Enforcing PATH_LEN
372  file = fopen(file_name,"wb"); //Open File
373  //Write when -and only WHEN - nothing went wrong data to file
374  if (file != 0) {
375  if(loglevel > 0) {
376  fprintf(stderr, "%s FILENAME: %s\n", log_time, file_name);
377  } else {
378  fprintf(stderr, "%s FILENAME: %s%s_%s_%s-%u_%u.ipm\n", log_time, \
379  data_path, log_time, ipv4icmp.dest_ip_str, "<masked by default loglevel>", ipv4icmp.type, ipv4icmp.code);
380  }
381  fwrite(ipv4icmp.data + data_offset, ipv4icmp.data_len - data_offset, 1, file);
382  CHECK(fflush(file), == 0);
383  fclose(file);
384  } else {
385  //if somthing went wrong, log it.
386  if(loglevel>0) {
387  fprintf(stderr, "%s ERROR: Could not write to file %s\n", log_time, file_name);
388  } else {
389  fprintf(stderr, "%s ERROR: Could not write to file %s%s_%s_%s-%u_%u.ipm\n", log_time, \
390  data_path, log_time, ipv4icmp.dest_ip_str, "<masked by default loglevel>", ipv4icmp.type, ipv4icmp.code);
391  }
392 
393  }
394  }
395 
396  //End time
397  long double duration = time_str(NULL, 0, stop_time, sizeof(stop_time)) - unix_timeasdouble; //...generate string with current time
398 
399  //Compute SHA1 of payload
400  SHA1(ipv4icmp.data, ipv4icmp.data_len, payload_sha1);
401  payload_sha1_str = print_hex_string(payload_sha1, SHA_DIGEST_LENGTH);
402  //Make HexDump output out of binary payload
403  payload_hd_str = hex_dump(ipv4icmp.data, ipv4icmp.data_len, true); //must be freed
404  payload_str = print_hex_string(ipv4icmp.data, ipv4icmp.data_len); //must be freed
405 
406  //Close ICMP JSON object with tainted status and "flow" part.
407  json_value.boolean = tainted;
408  dict_update(json_dict(false), JSON_BOOL, json_value, 2, "ICMP", "tainted");
409  json_value.string = log_time;
410  dict_update(json_dict(false), JSON_STR, json_value, 2, "FLOW", "start");
411  json_value.string = stop_time;
412  dict_update(json_dict(false), JSON_STR, json_value, 2, "FLOW", "end");
413  json_value.floating = duration;
414  dict_update(json_dict(false), JSON_FLOAT, json_value, 2, "FLOW", "duration");
415  json_value.integer = ipv4icmp.data_len;
416  dict_update(json_dict(false), JSON_INT, json_value, 2, "FLOW", "bytes_toserver");
417  json_value.string = payload_hd_str;
418  dict_update(json_dict(false), JSON_STR, json_value, 2, "FLOW", "payload_hd");
419  json_value.string = payload_str;
420  dict_update(json_dict(false), JSON_STR, json_value, 2, "FLOW", "payload_str");
421  json_value.string = payload_sha1_str;
422  dict_update(json_dict(false), JSON_STR, json_value, 2, "FLOW", "payload_sha1");
423 
424  //free str allocated by strndup() in function char *inttoa(uint32_t) and char *print_hex_string(const unsigned char*, unsigned int)
