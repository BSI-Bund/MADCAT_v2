44 //Helper function to parse IP Options. Returns tainted status, puts option data in hex string.
45 bool parse_ipopt(int opt_cpclno, const char* opt_name, \
46  unsigned char** opt_ptr_ptr, const unsigned char* beginofoptions_addr, const unsigned char* endofoptions_addr)
47 {
48  int opt_len = *(*opt_ptr_ptr+1);
49  if((*opt_ptr_ptr + opt_len) > endofoptions_addr) return true; //Check length, signal tainted in case of failure
50  //Option data to hex string
51  char* hex_string = print_hex_string(*opt_ptr_ptr + 2, opt_len - 2); //Extract option data as hex string. Has to be freed!
52  dict_update(json_dict(false), JSON_STR, json_value, 3, "IP", "ip_options", opt_name); //JSON Output
53  free(hex_string);
54  *opt_ptr_ptr += opt_len; //set pointer to next option
55  return false; //Option not tainted.
56 }

58 int analyze_ip_header(const unsigned char* packet, int recv_len)
59 {
60  if(recv_len - (IP_OR_TCP_HEADER_MINLEN) <= 0) return -1; //Malformed Paket
61 
62  struct iphdr *iphdr = (struct iphdr *)(packet);
63  char* ip_saddr = inttoa(iphdr->saddr); //Must be freed!
64  char* ip_daddr = inttoa(iphdr->daddr); //Must be freed!
65 
66  //printf("\n\nlength: %d\n version:%x\n TOS: %x\n tot_len: %d\nid: 0x%x\n flags: 0x%04x\n ttl: %d\n protocol: %d\n check: 0x%04x\n src_addr: %s dst_addr: %s\n\n",
67  json_value.integer = iphdr->ihl*4;
68  dict_update(json_dict(false), JSON_INT, json_value, 2, "IP", "hdr_len");
69  json_value.integer = iphdr->version;
70  dict_update(json_dict(false), JSON_INT, json_value, 2, "IP", "version");
71  json_value.hex.number = iphdr->tos;
72  json_value.hex.format = HEX_FORMAT_02;
73  dict_update(json_dict(false), JSON_HEX, json_value, 2, "IP", "tos");
74  json_value.integer = ntohs(iphdr->tot_len);
75  dict_update(json_dict(false), JSON_INT, json_value, 2, "IP", "tot_len");
76  json_value.hex.number = ntohs(iphdr->id);
77  json_value.hex.format = HEX_FORMAT_04;
78  dict_update(json_dict(false), JSON_HEX, json_value, 2, "IP", "id");
79  json_value.hex.number = ntohs(iphdr->frag_off);
80  json_value.hex.format = HEX_FORMAT_04;
81  dict_update(json_dict(false), JSON_HEX, json_value, 2, "IP", "flags");
82  json_value.integer = iphdr->ttl;
83  dict_update(json_dict(false), JSON_INT, json_value, 2, "IP", "ttl");
84  json_value.integer = iphdr->protocol;
85  dict_update(json_dict(false), JSON_INT, json_value, 2, "IP", "protocol");
86  json_value.hex.number = ntohs(iphdr->check);
87  json_value.hex.format = HEX_FORMAT_04;
88  dict_update(json_dict(false), JSON_HEX, json_value, 2, "IP", "checksum");
89  json_value.string = ip_saddr;
90  dict_update(json_dict(false), JSON_STR, json_value, 2, "IP", "src_addr");
91  json_value.string = ip_daddr;
92  dict_update(json_dict(false), JSON_STR, json_value, 2, "IP", "dest_addr");
93 
94  //Parse IP options
95  if (iphdr->ihl > 5) { //If Options/Padding present (IP Header longer than 5*4 = 20Byte)
96  bool eol = false; //EOL reached ?
97  bool tainted = false; //Is somethin unparsable inside the packet / options?
98  //calculate begin / end of options
99  const unsigned char* beginofoptions_addr = (unsigned char*) packet + ETHERNET_HEADER_LEN + IP_OR_TCP_HEADER_MINLEN;
100  const unsigned char* endofoptions_addr = packet + ETHERNET_HEADER_LEN + iphdr->ihl*4;
101  if(endofoptions_addr > (packet + recv_len)) {
102  //Malformed Paket
103  endofoptions_addr = (packet + recv_len); //Repair end of options address
104  tainted = true; //mark as tainted, thus do not parse, just dump hexstring
105  }
106  //set pointer to beginning of options
107  unsigned char* opt_ptr = (unsigned char*) beginofoptions_addr;
108  while(!tainted && !eol && opt_ptr < (packet + ETHERNET_HEADER_LEN + iphdr->ihl*4)) {
109  switch(*opt_ptr) {
110  case MY_IPOPT_EOOL: //EOL is only one byte, so this is hopefully going to be easy.
111  json_value.string = "";
112  dict_update(json_dict(false), JSON_STR, json_value, 3, "IP", "ip_options", "eol");
113  opt_ptr++;
114  eol = true;
115  break;
116  case MY_IPOPT_NOP: //NOP is only one byte, so this is going to be easy, too
117  json_value.string = "";
118  dict_update(json_dict(false), JSON_STR, json_value, 3, "IP", "ip_options", "nop");
119  opt_ptr++;

216 int analyze_udp_header(const unsigned char* packet, int recv_len)
217 {
218  struct iphdr *iphdr = (struct iphdr *)(packet);
219 
220  if(recv_len - (iphdr->ihl*4 + UDP_HEADER_LEN) <= 0) return -1; //Malformed Paket
221 
222  struct udphdr *udphdr = (struct udphdr *) (packet + iphdr->ihl*4);
223 
224  json_value.integer = ntohs(udphdr->source);
225  dict_update(json_dict(false), JSON_INT, json_value, 2, "UDP", "src_port");
226  json_value.integer = ntohs(udphdr->dest);
227  dict_update(json_dict(false), JSON_INT, json_value, 2, "UDP", "dest_port");
228  json_value.integer = ntohs(udphdr->len);
229  dict_update(json_dict(false), JSON_INT, json_value, 2, "UDP", "len");
230  json_value.integer = ntohs(udphdr->check);
231  dict_update(json_dict(false), JSON_INT, json_value, 2, "UDP", "checksum");
232  return 0;
233 }
